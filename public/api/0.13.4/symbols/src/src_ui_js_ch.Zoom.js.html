1 /**
  2 * Zoom shows a contextual reference to an augmented version of main declared image.
  3 * @name Zoom
  4 * @class Zoom
  5 * @augments ch.Floats
  6 * @requires ch.onImagesLoads
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {Boolean} [conf.fx] Enable or disable fade effect on show. By default, the effect are disabled.
 10 * @param {Boolean} [conf.context] Sets a reference to position of component that will be considered to carry out the position. By default is the anchor of HTML snippet.
 11 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or "lt rt" by default.
 12 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or "20 0" by default.
 13 * @param {String} [conf.content] This message will be shown when component needs to communicate that is in process of load. It's "Loading zoom..." by default.
 14 * @param {Number} [conf.width] Width of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 15 * @param {Number} [conf.height] Height of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 16 * @returns itself
 17 * @exampleDescription Create a Zoom component wrapping the original image with a anchor element pointing to a bigger version than the original.
 18 * @example
 19 * var widget = $(".example").zoom();
 20 * @factorized
 21 * @see ch.Floats
 22 * @see ch.Modal
 23 * @see ch.Tooltip
 24 * @see ch.Layer
 25 * @see ch.OnImagesLoads

244 		/**

246 		 * @private
247 		 * @name ch.Zoom#zoomed
248 		 * @type Object

250 			zoomed = (function () {
251 				// Define the content source
252 				var $img = $("<img src=\"" + that.element.href + "\" class=\"ch-hide\">").appendTo(that.$element);
253 
254 				if (isIE) { $img.css('visibility', 'hidden').removeClass('ch-hide'); }
255 
256 				// Grab some data when zoomed image loads
257 				$img.onImagesLoads(function () {
258 
259 					// Save the zoomed image size
260 					zoomed.width = $img.prop("width");
261 					zoomed.height = $img.prop("height");
262 
263 					if (isIE) { $img.css('visibility', 'visible').addClass('ch-hide'); }
264 
265 					that.content.configure({
266 						'input': $img
267 					});
268 
269 					// Save the zoom ratio
270 					ratio.width = zoomed.width / original.width;
271 					ratio.height = zoomed.height / original.height;
272 
273 					// Seeker: Size relative to zoomed image respect zoomed area
274 					var w = ~~(conf.width / ratio.width),
275 						h = ~~(conf.height / ratio.height);
276 
277 					// Seeker: Save half width and half height
278 					seeker.width = w / 2;
279 					seeker.height = h / 2;
280 
281 					// Seeker: Set size and append it
282 					seeker.$shape.css({"width": w, "height": h}).appendTo(that.$element);
283 
284 					// Remove loading
285 					$loading.remove();
286 
287 					// Change zoomed image status to Ready
288 					zoomed.ready = true;
289 
290 					// TODO: MAGIC here! if mouse is over image show seeker and make all that innerShow do
291 				});

332 		/**
333 		 * Calculates movement limits and sets it to seeker and augmented image.
334 		 * @private
335 		 * @function
336 		 * @name ch.Zoom#move
337 		 * @param {Event} event Mouse event to take the cursor position.
338 		 */
339 			move = function (event) {
340 
341 				var x, y;
342 
343 				// Left side of seeker LESS THAN left side of image
344 				if (event.pageX - seeker.width < offset.left) {
345 					x = 0;
346 				// Right side of seeker GREATER THAN right side of image
347 				} else if (event.pageX + seeker.width > original.width + offset.left) {
348 					x = original.width - (seeker.width * 2) - 2;
349 				// Free move
350 				} else {
351 					x = event.pageX - offset.left - seeker.width;
352 				}
353 
354 				// Top side of seeker LESS THAN top side of image
355 				if (event.pageY - seeker.height < offset.top) {
356 					y = 0;
357 				// Bottom side of seeker GREATER THAN bottom side of image
358 				} else if (event.pageY + seeker.height > original.height + offset.top) {
359 					y = original.height - (seeker.height * 2) - 2;
360 				// Free move
361 				} else {
362 					y = event.pageY - offset.top - seeker.height;
363 				}
364 
365 				// Move seeker
366 				seeker.$shape.css({"left": x, "top": y});
367 				
368 				// Move zoomed image
369 				zoomed.$image.css({"left": (-ratio.width * x), "top": (-ratio.height * y)});
370 
371 			};

377 		/**
378 		 * Inner show method. Attach the component's layout to the DOM tree and load defined content.
379 		 * @protected
380 		 * @name ch.Zoom#innerShow
381 		 * @function
382 		 * @returns itself
383 		 */
384 		that.innerShow = function () {
385 
386 			// If the component isn't loaded, show loading transition
387 			if (!zoomed.ready) {
388 				$loading.removeClass("ch-hide");
389 				return that;
390 			}
391 
392 			// Update position of anchor here because Zoom can be inside a Carousel and its position updates
393 			offset = that.$element.offset();
394 
395 			// Bind move calculations
396 			that.$element.bind("mousemove", function (event) { move(event); });
397 
398 			// Show seeker
399 			seeker.$shape.removeClass("ch-hide");
400 
401 			// Show float
402 			that.parent.innerShow();
403 
404 			return that;
405 		};

407 		/**
408 		 * Inner hide method. Hides the component's layout and detach it from DOM tree.
409 		 * @protected
410 		 * @name ch.Zoom#innerHide
411 		 * @function
412 		 * @returns itself
413 		 */
414 		that.innerHide = function () {
415 
416 			// If the component isn't loaded, hide loading transition
417 			if (!zoomed.ready) {
418 				$loading.addClass("ch-hide");
419 				return that;
420 			}
421 
422 			// Unbind move calculations
423 			that.$element.unbind("mousemove");
424 
425 			// Hide seeker
426 			seeker.$shape.addClass("ch-hide");
427 
428 			// Hide float
429 			that.parent.innerHide();
430 
431 			return that;
432 
433 		};

435 		/**
436 		 * Getter and setter for size attributes of float that contains the zoomed image.
437 		 * @protected
438 		 * @function
439 		 * @name ch.Zoom#size
440 		 * @param {string} prop Property that will be setted or getted, like "width" or "height".
441 		 * @param {string} [data] Only for setter. It's the new value of defined property.
442 		 * @returns itself
443 		 */
444 		that.size = function (prop, data) {
445 
446 			// Seeker: Updates styles and size value
447 			if (data) {
448 				// Seeker: Size relative to zoomed image respect zoomed area
449 				var size = ~~(data / ratio[prop]);
450 
451 				// Seeker: Save half width and half height
452 				seeker[prop] = size / 2;
453 
454 				// Seeker: Set size
455 				seeker.$shape.css(prop, size);
456 			}
457 
458 			// Change float size
459 			return that.parent.size(prop, data);
460 		};
