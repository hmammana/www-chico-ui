235 		/**
236 		 * Calculates and set the size of items and its margin to get an adaptive Carousel.
237 		 * @private
238 		 * @name ch.Carousel#updateDistribution
239 		 * @function
240 		 */
241 			updateDistribution = function () {
242 
243 				// Grabs if there are MORE THAN ONE item in a page or just one
244 				var moreThanOne = itemsPerPage > 1,
245 
246 				// Total space to use as margin into mask
247 				// It's the difference between mask width and total width of all items
248 					freeSpace = maskWidth - (itemOuterWidth * itemsPerPage),
249 
250 				// Width to add to each item to get responsivity
251 				// When there are more than one item, get extra width for each one
252 				// When there are only one item, extraWidth must be just the freeSpace
253 					extraWidth = moreThanOne ? Math.ceil(freeSpace / itemsPerPage / 2) : Math.ceil(freeSpace);
254 
255 				// Update ONLY IF margin changed from last redraw
256 				if (itemExtraWidth === extraWidth) { return; }
257 
258 				// Amount of spaces to distribute the free space
259 				// When there are 6 items on a page, there are 5 spaces between them
260 				// Except when there are only one page that NO exist spaces
261 				var spaces = moreThanOne ? itemsPerPage - 1 : 0,
262 
263 				// The new width calculated from current width plus extraWidth
264 					width = itemWidth + extraWidth;
265 
266 				// Update global value of width
267 				itemExtraWidth = extraWidth;
268 
269 				// Free space for each space between items
270 				// Ceil to delete float numbers (not Floor, because next page is seen)
271 				// There is no margin when there are only one item in a page
272 				// Update global values
273 				itemMargin = moreThanOne ? Math.ceil(freeSpace / spaces / 2) : 0;
274 
275 				// Update distance needed to move ONLY ONE page
276 				// The width of all items on a page, plus the width of all margins of items
277 				pageWidth = (itemOuterWidth + extraWidth + itemMargin) * itemsPerPage;
278 
279 				// Update the list width
280 				// Delete efects on list to change width instantly
281 				// Do it before item resizing to make space to all items
282 				$list.addClass("ch-carousel-nofx").css("width", pageWidth * pages);
283 
284 				// Restore efects to list if it's required
285 				// Use a setTimeout to be sure to do this after width change
286 				if (conf.fx) {
287 					setTimeout(function () { $list.removeClass("ch-carousel-nofx"); }, 0);
288 				}
289 
290 				// Update element styles
291 				// Get the height using new width and relation between width and height of item (ratio)
292 				$items.css({
293 					"width": width,
294 					"height": (width * itemHeight) / itemWidth,
295 					"margin-right": itemMargin
296 				});
297 
298 				// Update the mask height with the list height
299 				$mask.css("height", $list.outerHeight());
300 			},

302 		/**
303 		 * Trigger all recalculations to get the functionality measures.
304 		 * @private
305 		 * @name ch.Carousel#draw
306 		 * @function
307 		 */
308 			draw = function () {
309 
310                 // Save the current amount of items to know if restore the page after calculations
311                 var lastItemsPerPage = itemsPerPage,
312                     // Restore if itemsPerPage is the same after calculations
313                     lastPage = currentPage,
314                     // Restore if itemsPerPage is NOT the same after calculations (go to the current first item page)
315                     firstItemOnPage = ((currentPage - 1) * itemsPerPage) + 1,
316                     restoreFX = conf.fx;
317 
318                 $list.addClass("ch-carousel-nofx");
319                 conf.fx = false;
320 
321 				// Avoid wrong calculations going to first page
322 				goToPage(1);
323 
324 				// Trigger Draw event as deferred
325 				that.trigger("redraw");
326 
327 				// Trigger Draw event as configuration
328 				that.callbacks("onRedraw");
329 
330 				// Update the width of the mask
331 				maskWidth = $mask.outerWidth();
332 
333 				// Update amount of items into a single page (from conf or auto calculations)
334 				itemsPerPage = (function () {
335 					// The width of each item into the width of the mask
336 					var i = ~~(maskWidth / itemOuterWidth);
337 
338 					// Avoid zero items in a page
339 					if (i === 0) { return 1; }
340 
341 					// Limit amount of items when user set a maxItems amount
342 					if (ch.util.hasOwn(conf, "maxItems") && i > conf.maxItems) {
343 						return conf.maxItems;
344 					}
345 
346 					// Default calculation
347 					return i;
348 				}());
349 
350                 // Update the items collection
351                 $items = $list.children();
352 
353 				// Update amount of total pages
354 				// The ratio between total amount of items and items in each page
355 				var totalPages = Math.ceil(($items.length + queue.length) / itemsPerPage);
356 
357                 // Update only if pages amount changed from last redraw
358 				if (pages !== totalPages) {
359 					// Update value
360 					pages = totalPages;
361 					// Set WAI-ARIA properties to each item
362 					updateARIA();
363 					// Update arrows (when pages === 1, there is no arrows)
364 					updateArrows();
365 					// Update pagination
366 					updatePagination();
367 				}
368 
369                 // Analizes if next page needs to load items from queue and execute addItems() method
370                 loadAsyncItems();
371 
372 				// Update the margin between items and its size
373 				updateDistribution();
374 
375                 // Restore the page after calculations
376                 if (lastItemsPerPage === itemsPerPage) {
377                     goToPage(lastPage);
378                 // Restore the page of the first item before recalculations
379                 } else {
380                     goToPage(Math.ceil(firstItemOnPage / itemsPerPage));
381                 }
382 
383                 if (restoreFX) {
384                     setTimeout(function () { $list.removeClass("ch-carousel-nofx"); }, 0);
385                     conf.fx = restoreFX;
386                 }
387 			},

670 				// Collection of thumbnails strings

672 
673 				// Generate a thumbnail for each page on Carousel
674 				for (var i = 1, j = pages + 1; i < j; i += 1) {
675 
676 					// Determine if this thumbnail is selected or not

792 		/**
793 		 * Updates all necessary data to move to a specified page.
794 		 * @private
795 		 * @name ch.Carousel#goToPage
796 		 * @function
797 		 * @param {Number || String} page Reference of page to go. It can be specified as number or "first" or "last" string.
798 		 */
799 			goToPage = function (page) {
800 
801 				// Page getter
802 				if (!page) { return currentPage; }
803 
804 				// Page setter
805 				// Change to number the text pages ("first" and "last")
806 				if (page === "first") { page = 1; }
807 				else if (page === "last") { page = pages; }
808 
809 				// Avoid strings from here
810 				page = window.parseInt(page);
811 
812 				// Avoid to select:
813 				// - The same page that is selected yet
814 				// - A page less than 1
815 				// - A page greater than total amount of pages
816 				if (page === currentPage || page < 1 || page > pages) { return; }
817 
818 				// Perform these tasks in the following order:
819 				// Task 1: Move the list!!!
820 				// Position from 0 (zero), to page to move (page number beginning in zero)
821 				translate(-pageWidth * (page - 1));
822 
823 				// Task 2: Update selected thumbnail on pagination
824 				switchPagination(currentPage, page);
825 
826 				// Task 3: Update value of current page
827 				currentPage = page;
828 
829 				// Task 4: Check for arrows behavior on first, last and middle pages
830 				updateArrows();
831 
832 				// Task 5: Get items from queue to the list, if it's necessary
833 				loadAsyncItems();
834 
835 				// Task 6: Set WAI-ARIA properties to each item
836 				updateARIA();
837 
838 				// Trigger Select event as configuration
839 				that.trigger("select");
840 
841 				// Trigger Select event as deferred
842 				that.callbacks("onSelect");
843 			},

845 		/**
846 		 * Move items from queue to collection.
847 		 * @private
848 		 * @name ch.Carousel#addItems
849 		 * @function
850 		 * @param {Number} amount Amount of items that will be added.
851 		 */
852 			addItems = function (amount) {
853 
854 				// Take the sample from queue
855 				var sample = queue.splice(0, amount),
856 
857 				// Function with content processing using asyncRender or not
858 					getContent = conf.asyncRender || function (data) { return data; };
859 
860 				// Replace sample items with Carousel item template)
861 				for (var i = 0; i < amount; i += 1) {
862 					// Replace sample item
863 					sample[i] = [
864 						// Open tag with ARIA role
865 						"<li role=\"listitem\"",
866 						// Add classname to identify this as item
867 						" class=\"ch-carousel-item\"",
868 						// Add the same margin than all siblings items
869 						" style=\"width: " + (itemWidth + itemExtraWidth) + "px; margin-right: " + itemMargin + "px\"",
870 						// Add content (executing a template, if user specify it) and close the tag
871 						">" + getContent(sample[i]) + "</li>"
872 					// Get it as string
873 					].join("");
874 				};
875 
876 				// Add sample items to the list
877 				$list.append(sample.join(""));
878 
879 				// Update items collection
880 				$items = $list.children();
881 
882 				// Trigger item addition event as deferred
883 				that.trigger("itemsAdded");
884 
885 				// Trigger item addition event as configuration
886 				that.callbacks("onItemsAdded");
887 			},

889 		/**
890 		 * Analizes if next page needs to load items from queue and execute addItems() method.
891 		 * @private
892 		 * @name ch.Carousel#loadAsyncItems
893 		 * @function
894 		 */
895 			loadAsyncItems = function () {
896 
897 				// Load only when there are items in queue
898 				if (queue.length === 0) { return; }
899 
900 				// Amount of items from the beginning to current page
901 				var total = currentPage * itemsPerPage,
902 
903 				// How many items needs to add to items rendered to complete to this page
904 					amount = total - $items.length;
905 
906 				// Load only when there are items to add
907 				if (amount < 1) { return; }
908 
909 				// If next page needs less items than it support, then add that amount
910 				amount = (queue.length < amount) ? queue.length : amount;
911 
912 				// Add these
913 				addItems(amount);
914 			};
