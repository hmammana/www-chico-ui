1 /**
  2 * Datepicker lets you select dates.
  3 * @name DatePicker
  4 * @class DatePicker
  5 * @augments ch.Controls
  6 * @requires ch.Calendar
  7 * @see ch.Controls
  8 * @see ch.Calendar
  9 * @memberOf ch
 10 * @param {Object} [conf] Object with configuration properties.
 11 * @param {String} [conf.format] Sets the date format. By default is "DD/MM/YYYY".
 12 * @param {String} [conf.selected] Sets a date that should be selected by default. By default is the date of today.
 13 * @param {String} [conf.from] Set a maximum selectable date.
 14 * @param {String} [conf.to] Set a minimum selectable date.
 15 * @param {String} [conf.points] Points to be positioned. See Positioner component. By default is "ct cb".
 16 * @param {Array} [conf.monthsNames] By default is ["Enero", ... , "Diciembre"].
 17 * @param {Array} [conf.weekdays] By default is ["Dom", ... , "Sab"].
 18 * @param {Boolean} [conf.closable] Defines if floated component will be closed when a date is selected or not. By default it's "true".
 19 * @returns itself
 20 * @factorized
 21 * @exampleDescription Create a new datePicker.
 22 * @example
 23 * var widget = $(".example").datePicker();
 24 * @exampleDescription Create a new Date Picker with configuration.
 25 * @example
 26 * var widget = $(".example").datePicker({
 27 *	 "format": "MM/DD/YYYY",
 28 *	 "selected": "2011/12/25",
 29 *	 "from": "2010/12/25",
 30 *	 "to": "2012/12/25",
 31 *	 "monthsNames": ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"],
 32 *	 "weekdays": ["Su", "Mo", "Tu", "We", "Thu", "Fr", "Sa"]
 33 * });
 34 */
 35 (function (window, $, ch) {
 36 	'use strict';
 37 
 38 	if (window.ch === undefined) {
 39 		throw new window.Error('Expected ch namespace defined.');
 40 	}
 41 
 42 	var setTimeout = window.setTimeout,
 43 		setInterval = window.setInterval,
 44 		$document = $(window.document);
 45 
 46 	function DatePicker($el, conf) {
 47 
 48 		/**
 49 		* Reference to a internal component instance, saves all the information and configuration properties.
 50 		* @private
 51 		* @name ch.DatePicker#that
 52 		* @type object
 53 		*/
 54 		var that = this;
 55 
 56 		that.$element = $el;
 57 		that.element = $el[0];
 58 		that.type = 'datePicker';
 59 		conf = conf || {};
 60 
 61 		conf = ch.util.clone(conf);
 62 
 63 		// Configuration by default
 64 		conf.format = conf.format || "DD/MM/YYYY";
 65 		conf.points = conf.points || "ct cb";
 66 		conf.closable = ch.util.hasOwn(conf, "closable") ? conf.closable : true;
 67 
 68 		that.conf = conf;
 69 
 70 	/**
 71 	*	Inheritance
 72 	*/
 73 
 74 		that = ch.Controls.call(that);
 75 		that.parent = ch.util.clone(that);
 76 
 77 	/**
 78 	*	Private Members
 79 	*/
 80 
 81 	/**
 82 	*	Protected Members
 83 	*/
 84 
 85 		/**
 86 		* Pick a date in the Calendar and updates the input data.
 87 		* @protected
 88 		* @function
 89 		* @name ch.DatePicker#process
 90 		*/
 91 		that.process = function (event) {
 92 
 93 			// Day selection
 94 			if (event.target.nodeName !== "TD" || event.target.className.indexOf("ch-calendar-disabled") !== -1 || event.target.className.indexOf("ch-calendar-other") !== -1) {
 95 				return;
 96 			}
 97 
 98 			// Select the day and update input value with selected date
 99 
100 			that.element.value = that.calendar.selectDay(event.target.innerHTML);
101 
102 			// Hide float
103 			if (conf.closable) { that["float"].innerHide(); }
104 
105 			/**
106 			* Callback function
107 			* @public
108 			* @name ch.DatePicker#select
109 			* @event
110 			*/
111 			// Old callback system
112 			that.callbacks("onSelect");
113 			// New callback
114 			that.trigger("select");
115 
116 		};
117 
118 
119 		/**
120 		* Reference to the Calendar component instance.
121 		* @protected
122 		* @type Object
123 		* @name ch.DatePicker#calendar
124 		*/
125 		that.calendar = $("<div>")
126 			// Add functionality for date selection
127 			.on("click", function (event) { that.process(event); })
128 			// Instance Calendar component
129 			.calendar({
130 				"format": conf.format,
131 				"from": conf.from,
132 				"to": conf.to,
133 				"selected": conf.selected,
134 				"monthsNames": conf.monthsNames,
135 				"weekdays": conf.weekdays
136 			});
137 
138 		/**
139 		* Reference to the Float component instanced.
140 		* @protected
141 		* @type Object
142 		* @name ch.DatePicker#float
143 		*/
144 		that["float"] = that.createFloat({
145 			"$element": $("<i role=\"button\" class=\"ch-datePicker-trigger ch-icon-calendar\"></i>").insertAfter(that.element),
146 			"content": $(that.calendar.element),
147 			"points": conf.points,
148 			"offset": "-1 8",
149 			"aria": {
150 				"role": "tooltip"
151 			},
152 			"closeButton": false,
153 			"cone": true
154 		});
155 		// The input element is described by the float
156 		that.$element.attr('aria-describedby', 'ch-' + that.type + '-' + that['float'].uid)
157 
158 	/**
159 	*  Public Members
160 	*/
161 
162 		/**
163 		 * @borrows ch.Widget#uid as ch.Menu#uid
164 		 * @borrows ch.Widget#element as ch.Menu#element
165 		 * @borrows ch.Widget#type as ch.Menu#type
166 		 */
167 
168 		/**
169 		* Triggers the innerShow method and returns the public scope to keep method chaining.
170 		* @public
171 		* @name ch.DatePicker#show
172 		* @function
173 		* @returns itself
174 		* @exampleDescription Following the first example, using <code>widget</code> as modal's instance controller:
175 		* @example
176 		* widget.show();
177 		*/
178 		that["public"].show = function () {
179 
180 			that["float"].innerShow();
181 
182 			return that["public"];
183 		};
184 
185 		/**
186 		* Triggers the innerHide method and returns the public scope to keep method chaining.
187 		* @public
188 		* @name ch.DatePicker#show
189 		* @function
190 		* @returns itself
191 		* @exampleDescription Following the first example, using <code>widget</code> as modal's instance controller:
192 		* @example
193 		* widget.hide();
194 		*/
195 		that["public"].hide = function () {
196 			that["float"].innerHide();
197 
198 			return that["public"];
199 		};
200 
201 		/**
202 		* Select a specific date or returns the selected date.
203 		* @public
204 		* @since 0.9
205 		* @name ch.DatePicker#select
206 		* @function
207 		* @param {string} "YYYY/MM/DD".
208 		* @return itself
209 		*/
210 		that["public"].select = function (date) {
211 			// Select the day and update input value with selected date
212 			// Setter
213 			if (date) {
214 				that.calendar.select(date);
215 				that.element.value = that.calendar.select();
216 
217 				return that["public"];
218 			}
219 
220 			// Getter
221 			return that.calendar.select();
222 		};
223 
224 		/**
225 		* Returns date of today
226 		* @public
227 		* @since 0.9
228 		* @name ch.DatePicker#today
229 		* @function
230 		* @return date
231 		*/
232 		that["public"].today = function () {
233 			return that.calendar.today();
234 		};
235 
236 		/**
237 		* Move to the next month or year. If it isn't specified, it will be moved to next month.
238 		* @public
239 		* @function
240 		* @name ch.DatePicker#next
241 		* @param {String} time A string that allows specify if it should move to next month or year.
242 		* @return itself
243 		* @default Next month
244 		*/
245 		that["public"].next = function (time) {
246 			that.calendar.next(time);
247 
248 			return that["public"];
249 		};
250 
251 		/**
252 		* Move to the previous month or year. If it isn't specified, it will be moved to previous month.
253 		* @public
254 		* @function
255 		* @name ch.DatePicker#prev
256 		* @param {String} time A string that allows specify if it should move to previous month or year.
257 		* @return itself
258 		* @default Previous month
259 		*/
260 		that["public"].prev = function (time) {
261 			that.calendar.prev(time);
262 
263 			return that["public"];
264 		};
265 
266 		/**
267 		* Reset the Date Picker to date of today
268 		* @public
269 		* @name ch.DatePicker#reset
270 		* @function
271 		* @return itself
272 		*/
273 		that["public"].reset = function () {
274 
275 			// Delete input value
276 			that.element.value = "";
277 
278 			that.calendar.reset();
279 
280 			return that["public"];
281 		};
282 
283 		/**
284 		* Set a minimum selectable date.
285 		* @public
286 		* @name ch.DatePicker#from
287 		* @function
288 		* @param {string} "YYYY/MM/DD".
289 		* @return itself
290 		*/
291 		that["public"].from = function (date) {
292 			that.calendar.from(date);
293 
294 			return that["public"];
295 		};
296 
297 		/**
298 		* Set a maximum selectable date.
299 		* @public
300 		* @name ch.DatePicker#to
301 		* @function
302 		* @param {string} "YYYY/MM/DD".
303 		* @return itself
304 		*/
305 		that["public"].to = function (date) {
306 			that.calendar.to(date);
307 
308 			return that["public"];
309 		};
310 
311 
312 	/**
313 	*	Default event delegation
314 	*/
315 
316 		// Change type of input to "text"
317 		that.element.type = "text";
318 
319 		// Change value of input if there are a selected date
320 		that.element.value = (conf.selected) ? that.calendar.select() : that.element.value;
321 
322 		// Add show behaivor to float's trigger.
323 		that["float"].$element.on("click", function (event) {
324 			that["float"].innerShow(event);
325 		});
326 
327 		/**
328 		* Triggers when the component is ready to use (Since 0.8.0).
329 		* @name ch.DatePicker#ready
330 		* @event
331 		* @public
332 		* @exampleDescription Following the first example, using <code>widget</code> as Date Picker's instance controller:
333 		* @example
334 		* widget.on("ready", function () {
335 		* 	this.show();
336 		* });
337 		*/
338 		setTimeout(function () { that.trigger("ready"); }, 50);
339 
340 		return that['public'];
341 	}
342 
343 	DatePicker.prototype.name = 'datePicker';
344 	DatePicker.prototype.constructor = DatePicker;
345 
346 	ch.factory(DatePicker);
347 
348 }(this, this.jQuery, this.ch));
