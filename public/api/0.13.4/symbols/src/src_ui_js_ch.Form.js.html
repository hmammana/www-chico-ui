 44 		that.type = 'form';

 48 		* Validation
 49 		*/
 50 		// Are there action and submit type?
 51 		if ( that.$element.find(":submit").length == 0 || that.$element.attr("action") == "" ){

 54 		};

 58 			for(var i = 0, j = ch.instances.form.length; i < j; i+=1){
 59 				if(ch.instances.form[i].element === that.element){
 60 					return {
 61 						exists: true,
 62 						object: ch.instances.form[i]
 63 					};
 64 				}
 65 			};
 66 		}

 71 		// Grab submit button
 72 		that.$submit = that.$element.find("input:submit");
 73

 77 	*  Inheritance
 78 	*/

 87 		/**
 88 		* A Boolean property that indicates is exists errors in the form.
 89 		* @private
 90 		* @name ch.Form#status
 91 		* @type boolean
 92 		*/
 93 		var status = true;
 94 
 95 		/**
 96 		* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
 97 		*/
 98 		var validate = function (event) {
 99 
100 			/**
101 			* Fired before the validations engine start.
102 			* @name ch.Form#beforeValidate
103 			* @event
104 			* @public
105 			* @exampleDescription
106 			* @example
107 			* widget.on("beforeValidate",function () {
108 			*	sowidget.action();
109 			* });
110 			*/
111 			that.callbacks("beforeValidate");
112 			// new callback
113 			that.trigger("beforeValidate");
114 
115 			// Status OK (with previous error)
116 			if ( !status ) {
117 				status = true;
118 			};
119 
120 			var i = 0, j = that.children.length, toFocus, childrenError = [];
121 
122 			// Shoot validations
123 			for (i; i < j; i+=1) {
124 				var child = that.children[i];
125 
126 				// Validate
127 				// Save children with errors
128 				if ( child.hasError() ) {
129 					childrenError.push(child);
130 				}
131 			};
132 
133 			// Is there's an error
134 			if (childrenError.length > 0) {
135 				status = false;
136 				// Issue UI-332: On validation must focus the first field with errors.
137 				// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
138 				if (childrenError[0].element.tagName === "DIV") {
139 					$(childrenError[0].element).find("input:first").focus();
140 				} else if (childrenError[0].element.type !== "hidden") {
141 					childrenError[0].element.focus();
142 				}
143 			} else {
144 				status = true;
145 			}
146 
147 			/**
148 			* Fired when the form validates.
149 			* @name ch.Form#validate
150 			* @event
151 			* @public
152 			* @exampleDescription
153 			* @example
154 			* widget.on("validate",function () {
155 			*	sowidget.action();
156 			* });
157 			*/
158 
159 			/**
160 			* Fired when the form fall on a error.
161 			* @name ch.Form#error
162 			* @event
163 			* @public
164 			* @exampleDescription
165 			* @example
166 			* widget.on("error",function () {
167 			*	sowidget.action();
168 			* });
169 			*/
170 			if (status) {
171 				that.callbacks("onValidate");
172 				// new callback
173 				that.trigger("validate");
174 			} else {
175 				that.callbacks("onError");
176 				// new callback
177 				that.trigger("error");
178 			}
179 
180 			/**
181 			* Fired when the validations end.
182 			* @name ch.Form#afterValidate
183 			* @event
184 			* @public
185 			* @exampleDescription
186 			* @example
187 			* widget.on("afterValidate",function () {
188 			*	sowidget.action();
189 			* });
190 			*/
191 			that.callbacks("afterValidate");
192 			// new callback
193 			that.trigger("afterValidate");
194 
195 			return that;
196 		};
197 
198 		/**
199 		* This methods triggers the 'beforSubmit' callback, then will execute validate() method,
200 		* and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
201 		*/
202 		var submit = function(event) {
203 
204 			/**
205 			* Fired before the form's submition.
206 			* @name ch.Form#beforeSubmit
207 			* @event
208 			* @public
209 			* @exampleDescription
210 			* @example
211 			* widget.on("beforeSubmit",function () {
212 			*	sowidget.action();
213 			* });
214 			*/
215 			that.callbacks("beforeSubmit");
216 			// new callback
217 			that.trigger("beforeSubmit");
218 
219 			// Execute all validations
220 			validate(event);
221 
222 			// If an error occurs prevent default actions
223 			if (!status) {
224 				ch.util.prevent(event);
225 		        if (event) {
226 		            event.stopImmediatePropagation();
227 		        }
228 			}
229 
230 			// OLD CALLBACK SYSTEM!
231 			// Is there's no error but there's a onSubmit callback
232 			if ( status && ch.util.hasOwn(conf, "onSubmit")) {
233 				// Avoid default actions
234 				ch.util.prevent(event);
235 				// To execute defined onSubmit callback
236 				that.callbacks("onSubmit");
237 			}
238 
239 			/**
240 			* Fired when submits the form.
241 			* @name ch.Form#submit
242 			* @event
243 			* @public
244 			* @exampleDescription
245 			* @example
246 			* widget.on("afterSubmit",function () {
247 			*	widget.action();
248 			* });
249 			*/
250 			// * New callback system *
251 			// Check inside $.data if there's a handler for ch-submit event
252 			// if something found there, avoid submit.
253 
254 			var formEvents = $(that["public"]).data("events");
255 			var isSubmitEventDefined = (formEvents && ch.util.hasOwn(formEvents, "ch-submit"));
256 
257 			if (status && isSubmitEventDefined){
258 				// Avoid default actions
259 				ch.util.prevent(event);
260 				// new callback
261 				that.trigger("submit");
262 			};
263 
264 			/**
265 			* Fired after the form's submition.
266 			* @name ch.Form#afterSubmit
267 			* @event
268 			* @public
269 			* @exampleDescription
270 			* @example
271 			* widget.on("afterSubmit",function () {
272 			*	this.reset();
273 			* });
274 			*/
275 			that.callbacks("afterSubmit");
276 			// new callback
277 			that.trigger("afterSubmit");
278 
279 			// Return that to chain methods
280 			return that;
281 		};
282 
283 		/**
284 		* Use this method to clear al validations.
285 		*/
286 		var clear = function(){
287 
288 			var i = 0, j = that.children.length;
289 			for(i; i < j; i += 1) {
290 				that.children[i].clear();
291 			}
292 
293 			status = true;
294 
295 			/**
296 			* Fired when clean the form's data.
297 			* @name ch.Form#clear
298 			* @event
299 			* @public
300 			* @exampleDescription
301 			* @example
302 			* widget.on("clear",function () {
303 			*	this.reset();
304 			* });
305 			*/
306 			that.callbacks("onClear");
307 			// new callback
308 			that.trigger("clear");
309 
310 			return that;
311 		};

314 		* Use this method to reset the form's input elements.

319

321 			* Fired when resets the form.
322 			* @name ch.Form#reset
323 			* @event
324 			* @public
325 			* @exampleDescription
326 			* @example
327 			* widget.on("reset",function () {
328 			*	sowidget.action();
329 			* });
330 			*/
331 			that.callbacks("onReset");
332 			// new callback
333 			that.trigger("reset");
334 
335 			return that;
336 		};

411 			return status;
412 		};

418 		* @function

437

439 		};

442 	*  Default event delegation
443 	*/

447 			that.$element.bind('submit', function(event){ ch.util.prevent(event); });
448 			// Delete all click handlers asociated to submit button >NATAN: Why?
449 				//Because if you want to do something on submit, you need that the trigger (submit button)
450 				//don't have events associates. You can add funcionality on onSubmit callback
451 			that.$element.find(":submit").unbind('click');
452 		};
453 
454 		// Bind the submit
455 		that.$element.bind("submit", function(event)Â { submit(event) });
456 
457 		// Bind the reset
458 		that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
459 
460 		// Listen the event "validate" from validations
461 		that.on('validated', function () {
462 			status = true;
463 		});

466 		* Triggers when the component is ready to use (Since 0.8.0).
467 		* @name ch.Form#ready
468 		* @event
469 		* @public
470 		* @since 0.8.0
471 		* @exampleDescription Following the first example, using <code>widget</code> as form's instance controller:
472 		* @example
473 		* widget.on("ready",function () {
474 		*	this.reset();
475 		* });
476 		*/
477 		window.setTimeout(function(){ that.trigger("ready")}, 50);
478 
479 		return that['public'];
