1 /**
  2 * Countdown counts the maximum of characters that user can enter in a form control. Countdown could limit the possibility to continue inserting charset.
  3 * @name Countdown
  4 * @class Countdown
  5 * @augments ch.Controls
  6 * @see ch.Controls
  7 * @memberOf ch
  8 * @param {Object} conf Object with configuration properties.
  9 * @param {Number} conf.max Number of the maximum amount of characters user can input in form control.
 10 * @param {String} [conf.plural] Message of remaining amount of characters, when it's different to 1. The variable that represents the number to be replaced, should be a hash. By default this parameter is "# characters left.".
 11 * @param {String} [conf.singular] Message of remaining amount of characters, when it's only 1. The variable that represents the number to be replaced, should be a hash. By default this parameter is "# character left.".
 12 * @returns itself
 13 * @factorized
 14 * @exampleDescription Create a simple Countdown. Then <code>widget</code> is a reference to the Countdown instance controller.
 15 * @example
 16 * var widget = $(".some-form-control").countdown(500);
 17 * @exampleDescription Create a Countdown with configuration.
 18 * @example
 19 * var widget = $(".some-form-control").countdown({
 20 *     "max": 500,
 21 *     "plural": "Restan # caracteres.",
 22 *     "singular": "Resta # caracter."
 23 * });
 24 */
 25 (function (window, $, ch) {
 26 	'use strict';
 27 
 28 	if (window.ch === undefined) {
 29 		throw new window.Error('Expected ch namespace defined.');
 30 	}
 31 
 32 	var setTimeout = window.setTimeout;
 33 
 34 	function Countdown($el, conf) {
 35 
 36 		/**
 37 		* Reference to an internal component instance, saves all the information and configuration properties.
 38 		* @private
 39 		* @name ch.Countdown#that
 40 		* @type Object
 41 		*/
 42 		var that = this;
 43 
 44 		that.$element = $el;
 45 		that.element = $el[0];
 46 		that.type = 'countdown';
 47 		conf = conf || {};
 48 
 49 		conf = ch.util.clone(conf);
 50 
 51 		// Configuration by default
 52 		// Max length of content
 53 		conf.max = parseInt(conf.max) || conf.num ||Â parseInt(conf.content) || 500;
 54 
 55 		// Messages
 56 		conf.plural = conf.plural || "# characters left.";
 57 		conf.singular = conf.singular || "# character left.";
 58 
 59 		that.conf = conf;
 60 
 61 	/**
 62 	*	Inheritance
 63 	*/
 64 
 65 		that = ch.Controls.call(that);
 66 		that.parent = ch.util.clone(that);
 67 
 68 	/**
 69 	*	Private Members
 70 	*/
 71 		/**
 72 		* Length of value of form control.
 73 		* @private
 74 		* @name ch.Countdown#contentLength
 75 		* @type Number
 76 		*/
 77 		var contentLength = that.element.value.length,
 78 
 79 		/**
 80 		* Amount of free characters until full the field.
 81 		* @private
 82 		* @name ch.Countdown#remaining
 83 		* @type Number
 84 		*/
 85 			remaining = conf.max - contentLength,
 86 
 87 		/**
 88 		* Change the visible message of remaining characters.
 89 		* @private
 90 		* @name ch.Countdown#updateRemaining
 91 		* @function
 92 		* @param num {Number} Remaining characters.
 93 		*/
 94 			updateRemaining = (function () {
 95 
 96 				// Singular or Plural message depending on amount of remaining characters
 97 				var message = (remaining === 1) ? conf.singular : conf.plural,
 98 
 99 				// Append to container to allow icon aside inputs
100 					$container = that.$element.parent(),
101 
102 				// Create the DOM Element when message will be shown
103 					$display = $("<p class=\"ch-form-hint\">" + message.replace("#", remaining) + "</p>").appendTo($container);
104 
105 				// Real function
106 				return function (num) {
107 
108 					// Singular or Plural message depending on amount of remaining characters
109 					var message = (num !== 1 ? conf.plural : conf.singular).replace(/\#/g, num);
110 
111 					// Update DOM text
112 					$display.text(message);
113 
114 					// Update amount of remaining characters
115 					remaining = num;
116 
117 				};
118 
119 			}());
120 
121 	/**
122 	*	Protected Members
123 	*/
124 
125 		/**
126 		* Process input of data on form control and updates remaining amount of characters or limits the content length
127 		* @protected
128 		* @name ch.Countdown#process
129 		* @function
130 		*/
131 		that.process = function () {
132 
133 			var len = that.element.value.length;
134 
135 			// Countdown or Countup
136 			if ((len > contentLength && len <= conf.max) || (len < contentLength && len >= 0)) {
137 
138 				// Change visible message of remaining characters
139 				updateRemaining(remaining - (len - contentLength));
140 
141 				// Update length of value of form control.
142 				contentLength = len;
143 
144 			// Limit Count
145 			} else if (len > contentLength && len > conf.max) {
146 
147 				// Cut the string value of form control
148 				that.element.value = that.element.value.substr(0, conf.max);
149 
150 			};
151 
152 		};
153 
154 
155 	/**
156 	*	Public Members
157 	*/
158 
159 		/**
160 		 * @borrows ch.Widget#uid as ch.Menu#uid
161 		 * @borrows ch.Widget#element as ch.Menu#element
162 		 * @borrows ch.Widget#type as ch.Menu#type
163 		 */
164 
165 	/**
166 	*	Default event delegation
167 	*/
168 
169 		// Bind process function to element
170 		that.$element.on("keyup keypress paste", function () { setTimeout(that.process, 0); });
171 
172 		/**
173 		* Triggers when component is ready to use.
174 		* @name ch.Countdown#ready
175 		* @event
176 		* @public
177 		* @exampleDescription Following the first example, using <code>widget</code> as Countdown's instance controller:
178 		* @example
179 		* widget.on("ready",function () {
180 		*	this.element;
181 		* });
182 		*/
183 		setTimeout(function () { that.trigger("ready"); }, 50);
184 
185 		return that['public'];
186 	};
187 
188 	Countdown.prototype.name = 'countdown';
189 	Countdown.prototype.constructor = Countdown;
190 
191 	ch.factory(Countdown);
192 
193 }(this, this.jQuery, this.ch));
