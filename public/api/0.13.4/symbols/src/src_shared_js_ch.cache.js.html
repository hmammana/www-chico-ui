1 /**
  2  * Cache control utility.
  3  * @name Cache
  4  * @class Cache
  5  * @memberOf ch
  6  */
  7 (function (window, ch) {
  8 	'use strict';
  9 
 10 	if (ch === undefined) {
 11 		throw new window.Error('Expected ch namespace defined.');
 12 	}
 13 
 14 	// Cache control utility.
 15 	var cache = {};
 16 
 17 	/**
 18 	* Map of cached resources
 19 	* @public
 20 	* @name ch.Cache#map
 21 	* @type object
 22 	*/
 23 	cache.map = {};
 24 
 25 	/**
 26 	* Set a resource to the cache control
 27 	* @public
 28 	* @function
 29 	* @name ch.Cache#set
 30 	* @param {string} url Resource location
 31 	* @param {string} data Resource information
 32 	*/
 33 	cache.set = function (url, data) {
 34 		cache.map[url] = data;
 35 	};
 36 
 37 	/**
 38 	* Get a resource from the cache
 39 	* @public
 40 	* @function
 41 	* @name ch.Cache#get
 42 	* @param {string} url Resource location
 43 	* @returns data Resource information
 44 	*/
 45 	cache.get = function (url) {
 46 		return cache.map[url];
 47 	};
 48 
 49 	/**
 50 	* Remove a resource from the cache
 51 	* @public
 52 	* @function
 53 	* @name ch.Cache#rem
 54 	* @param {string} url Resource location
 55 	*/
 56 	cache.rem = function (url) {
 57 		cache.map[url] = null;
 58 		delete cache.map[url];
 59 	};
 60 
 61 	/**
 62 	* Clears the cache map
 63 	* @public
 64 	* @function
 65 	* @name ch.Cache#flush
 66 	*/
 67 	cache.flush = function () {
 68 		delete cache.map;
 69 		cache.map = {};
 70 	};
 71 
 72 	ch.cache = cache;
 73 
 74 }(this, this.ch));
