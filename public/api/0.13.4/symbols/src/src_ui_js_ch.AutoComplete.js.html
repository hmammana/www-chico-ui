1 /**
  2 * AutoComplete lets you suggest anything from an input element. Use a suggestion service or use a collection with the suggestions.
  3 * @name AutoComplete
  4 * @class AutoComplete
  5 * @augments ch.Controls
  6 * @see ch.Controls
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {String} conf.url The url pointing to the suggestions's service.
 10 * @param {String} [conf.content] It represent the text when no options are shown.
 11 * @param {Array} [conf.suggestions] The suggestions's collection. If a URL is set at conf.url parametter this will be omitted.
 12 * @returns itself
 13 * @factorized
 14 * @exampleDescription Create a new autoComplete with configuration.
 15 * @example
 16 * var widget = $(".example").autoComplete({
 17 *     "url": "http://site.com/mySuggestions?q=",
 18 *     "message": "Write..."
 19 * });

 68 		/**
 69 		* Select an item.
 70 		* @private
 71 		* @type Function
 72 		* @name ch.AutoComplete#selectItem
 73 		*/
 74 		var selectItem = function (arrow, event) {
 75 			ch.util.prevent(event);
 76 
 77 			if (that.selected === (arrow === "bottom" ? that.items.length - 1 : 0)) { return; }
 78 			$(that.items[that.selected]).removeClass("ch-autoComplete-selected");
 79 
 80 			if (arrow === "bottom") { that.selected += 1; } else { that.selected -= 1; }
 81 			$(that.items[that.selected]).addClass("ch-autoComplete-selected");
 82 		};

 94 		that.selected = -1;

 97 		* List of the shown suggestions.
 98 		* @protected
 99 		* @type Array

154 		/**
155 		* It sets On/Off the loading icon.
156 		* @protected
157 		* @function
158 		* @name ch.AutoComplete#loading
159 		*/
160 		that.loading = function(show){
161 			if(show){
162 				that.$element.addClass("ch-autoComplete-loading");
163 			} else {
164 				that.$element.removeClass("ch-autoComplete-loading");
165 			}
166 		}

168 		/**
169 		* It fills the content inside the element represented by the float.
170 		* @protected
171 		* @function
172 		* @name ch.AutoComplete#populateContent
173 		*/
174 		that.populateContent = function (event,result) {
175 			// No results doesn't anything
176 			if (result.length===0 || that.element.value==="") {
177 				that.loading(false);
178 				that["float"].innerHide();
179 				return that;
180 			}
181 
182 			// Only one result and the same as the input hide float and doesn't anything
183 			if (result.length===1 && result[0]===that.element.value) {
184 				that.loading(false);
185 				that["float"].innerHide();
186 				return that;
187 			}
188 
189 			var list = "";
190 			$.each(result, function (i, e) {
191 				list+="<li data-index=\""+i+"\">"+e+"</li>";
192 			})
193 
194 			that.trigger("contentUnload");
195 			that.$content.html(list);
196 			that.selected = -1;
197 
198 			that["float"].content.configure({
199 				'input': that.$content
200 			});
201 
202 			that.trigger("contentLoaded");
203 
204 			that.items = that.$content.children();
205 
206 			// Adds only once the behavior
207 			if (!that.behaviorActived) {
208 				that.suggestionsBehavior(event);
209 				that.behaviorActived = true;
210 			}
211 
212 			that["float"].innerShow();
213 			that.loading(false);
214 			return that;
215 		}
216 
217 		/**
218 		* It does the query to the server if configured an URL, or it does the query inside the array given.
219 		* @protected
220 		* @function
221 		* @name ch.AutoComplete#doQuery
222 		*/
223 		that.doQuery = function(event){
224 
225 			if(!event){
226 				that.populateContent(window.event,that.suggestions);
227 				return that;
228 			}
229 
230 			var q = that.$element.val().toLowerCase();
231 			// When URL is configured it will execute an ajax request.
232 			if (that.element.value !== "" && event.keyCode !== 38 && event.keyCode !== 40  && event.keyCode !== 13  && event.keyCode !== 27) {
233 				if (that.conf.url !== undefined) {
234 					that.loading(true);
235 					var result = $.ajax({
236 						url: that.conf.url + q + "&callback=" + that.conf.jsonpCallback,
237 						dataType:"jsonp",
238 						cache:false,
239 						global:true,
240 						context: window,
241 						jsonp:that.conf.jsonpCallback,
242 						crossDomain:true
243 					});
244 				// When not URL configured and suggestions array were configured it search inside the suggestions array.
245 				} else if (that.conf.url === undefined) {
246 					var result = [];
247 					for(var a=(that.suggestions.length-1);(a+1);a--){
248 						var word = that.suggestions[a].toLowerCase();
249 						var exist = word.search(q);
250 						if(!exist){
251 							result.push(that.suggestions[a]);
252 						}
253 					};
254 					that.populateContent(event,result);
255 				}
256 			}
257 			return that;
258 		}
259 
260 		/**
261 		* Binds the behavior related to the list.
262 		* @protected
263 		* @function
264 		* @name ch.AutoComplete#suggestionsBehavior
265 		*/
266 		that.suggestionsBehavior = function(event){
267 			// BACKSPACE key bheavior. When backspace go to the start show the message
268 			$document.on(ch.events.key.BACKSPACE + ".autoComplete", function (x, event) {
269 
270 				// When the user make backspace with empty input autocomplete is shutting off
271 				if(that.element.value.length===0){
272 					ch.util.prevent(event);
273 					that.$element.trigger("blur");
274 				}
275 
276 				// When isn't any letter it hides the float
277 				if(that.element.value.length<=1){
278 					that["float"].innerHide();
279 					that.loading(false);
280 				}
281 
282 			})
283 			// ESC key behavior, it closes the suggestions's list
284 			.on(ch.events.key.ESC + ".autoComplete", function (x, event) { that.$element.trigger("blur"); })
285 			// ENTER key behavior, it selects the item who is selected
286 			.on(ch.events.key.ENTER + ".autoComplete", function (x, event) { that.$element.val($(that.items[that.selected]).text()); that.$element.trigger("blur"); })
287 			// UP ARROW key behavior, it selects the previous item
288 			.on(ch.events.key.UP_ARROW + ".autoComplete", function (x, event) { selectItem("up", event); })
289 			// DOWN ARROW key behavior, it selects the next item
290 			.on(ch.events.key.DOWN_ARROW + ".autoComplete", function (x, event) { selectItem("bottom", event); });
291 			// MouseOver & MouseDown Behavior
292 			that["float"].$content.on("mouseover mousedown",function(evt){
293 				var event = evt || window.event;
294 				var target = event.target || event.srcElement;
295 				var type = event.type;
296 				if(target.tagName === "LI"){
297 					// mouse over behavior
298 					if(type === "mouseover"){
299 						// removes the class if one is selected
300 						$(that.items[that.selected]).removeClass("ch-autoComplete-selected");
301 						// selects the correct item
302 						that.selected = parseInt(target.getAttribute("data-index"));
303 						// adds the class to highlight the item
304 						$(that.items[that.selected]).addClass("ch-autoComplete-selected");
305 					}
306 					// mouse down behavior
307 					if(type === "mousedown") {
308 						ch.util.prevent(event);
309 						that.$element.val($(that.items[that.selected]).text());
310 						that.$element.trigger("blur");
311 					}
312 				}
313 			});
314 		}
