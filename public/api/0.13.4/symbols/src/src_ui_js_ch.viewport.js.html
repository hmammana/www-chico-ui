1 /**
  2  * Viewport is a reference to position and size of the visible area of browser.
  3  * @name Viewport
  4  * @class Viewport
  5  * @standalone
  6  * @memberOf ch
  7  */
  8 (function (window, $, ch) {
  9 	'use strict';
 10 
 11 	if (ch === undefined) {
 12 		throw new window.Error('Expected ch namespace defined.');
 13 	}
 14 
 15 	var $window = $(window),
 16 		// Viewport is a reference to position and size of the visible area of browser.
 17 		viewport = {};
 18 
 19 	/**
 20 	 * Width of the visible area.
 21 	 * @public
 22 	 * @name ch.Viewport#width
 23 	 * @type Number
 24 	 */
 25 	viewport.width = $window.width();
 26 
 27 	/**
 28 	 * Height of the visible area.
 29 	 * @public
 30 	 * @name ch.Viewport#height
 31 	 * @type Number
 32 	 */
 33 	viewport.height = $window.height();
 34 
 35 	/**
 36 	 * Left offset of the visible area.
 37 	 * @public
 38 	 * @name ch.Viewport#left
 39 	 * @type Number
 40 	 */
 41 	viewport.left = $window.scrollLeft();
 42 
 43 	/**
 44 	 * Top offset of the visible area.
 45 	 * @public
 46 	 * @name ch.Viewport#top
 47 	 * @type Number
 48 	 */
 49 	viewport.top = $window.scrollTop();
 50 
 51 	/**
 52 	 * Right offset of the visible area.
 53 	 * @public
 54 	 * @name ch.Viewport#right
 55 	 * @type Number
 56 	 */
 57 	viewport.right = $window.scrollLeft() + $window.width();
 58 
 59 	/**
 60 	 * Bottom offset of the visible area.
 61 	 * @public
 62 	 * @name ch.Viewport#bottom
 63 	 * @type Number
 64 	 */
 65 	viewport.bottom = $window.scrollTop() + $window.height();
 66 
 67 	/**
 68 	 * Element representing the visible area.
 69 	 * @public
 70 	 * @name ch.Viewport#element
 71 	 * @type Object
 72 	 */
 73 	viewport.element = $window;
 74 
 75 	/**
 76 	 * Updates width and height of the visible area and updates ch.viewport.width and ch.viewport.height
 77 	 * @public
 78 	 * @function
 79 	 * @name ch.Viewport#getSize
 80 	 * @returns Object
 81 	 */
 82 	viewport.getSize = function () {
 83 
 84 		return {
 85 			"width": ch.viewport.width = $window.width(),
 86 			"height": ch.viewport.height = $window.height()
 87 		};
 88 	};
 89 
 90 	/**
 91 	 * Updates left, top, right and bottom coordinates of the visible area, relative to the window.
 92 	 * @public
 93 	 * @function
 94 	 * @name ch.Viewport#getPosition
 95 	 * @returns Object
 96 	 */
 97 	viewport.getPosition = function () {
 98 
 99 		var size = viewport.getSize();
100 
101 		return {
102 			"left": 0,
103 			"top": 0,
104 			"right": size.width,
105 			"bottom": size.height,
106 			// Size is for use as context on Positioner
107 			// (see getCoordinates method on Positioner)
108 			"width": size.width,
109 			"height": size.height
110 		};
111 	};
112 
113 	/**
114 	 * Updates left, top, right and bottom coordinates of the visible area, relative to the document.
115 	 * @public
116 	 * @function
117 	 * @name ch.Viewport#getOffset
118 	 * @returns Object
119 	 */
120 	viewport.getOffset = function () {
121 
122 		var position = viewport.getPosition(),
123 			scrollLeft = $window.scrollLeft(),
124 			scrollTop = $window.scrollTop();
125 
126 		return {
127 			"left": ch.viewport.left = scrollLeft,
128 			"top": ch.viewport.top = scrollTop,
129 			"right": ch.viewport.right = scrollLeft + position.right,
130 			"bottom": ch.viewport.bottom = scrollTop + position.bottom
131 		};
132 	};
133 
134 	ch.viewport = viewport;
135 
136 }(this, this.jQuery, this.ch));
