 63 		 * @name ch.Floats#onmessage
 64 		 * @function
 65 		 * @returns {this}
 66 		 */
 67 		that.content.onmessage = function (data) {
 68 
 69 			if (!that.active) { return; }
 70 
 71 			that.$content.html(data);
 72 
 73 			that.trigger("contentLoad");
 74 			if (ch.util.hasOwn(conf, "onContentLoad")) {
 75 				conf.onContentLoad.call((that.controller || that), data);
 76 			}
 77 
 78 			that.position('refresh');
 79 		};

 82 		 * This callback is triggered when async request fails.
 83 		 * @protected
 84 		 * @name ch.Floats#onerror
 85 		 * @function
 86 		 * @returns {this}
 87 		 */
 88 		that.content.onerror = function (data) {
 89 
 90 			if (!that.active) { return; }
 91 
 92 			that.$content.html(data);
 93 
 94 			that.trigger("contentError");
 95 			if (ch.util.hasOwn(conf, "onContentError")) {
 96 				conf.onContentError.call((that.controller || that), data.jqXHR, data.textStatus, data.errorThrown);
 97 			}
 98 
 99 			that.position('refresh');
100 		};

123 		 * Closable behavior.
124 		 * @private
125 		 * @function
126 		 * @name ch.Floats-closable
127 		 */
128 		// TODO: Create "closable" interface
129 		var closable = (function () {
130 			/**
131 			 * Returns any if the component closes automatic.
132 			 * @public
133 			 * @function
134 			 * @methodOf ch.Floats#closable
135 			 * @exampleDescription to get the height
136 			 * @example
137 			 * widget.closable() // true | false | "button"
138 			 * @returns boolean | string
139 			 */
140 			that["public"].closable = function () {
141 				return that.closable;
142 			};
143 
144 
145 			return function () {
146 
147 				// Closable Off: don't anything
148 				if (!that.closable) { return; }
149 
150 				// Closable On
151 				if (ch.util.hasOwn(conf, "closeButton") && conf.closeButton || ch.util.hasOwn(conf, "event") && conf.event === "click") {
152 					// Append close buttons
153 					// It will close with close button
154                     $('<a class="ch-close" role="button" style="z-index:' + (ch.util.zIndex += 1) + '"></a>')
155                         .on('click', function (event) {
156                             that.innerHide(event);
157                         })
158                         .prependTo(that.$container);
159 				}
160 
161 				// ESC key support
162 				$document.on(ch.events.key.ESC, function () {
163 					that.innerHide();
164 				});
165 
166 				// It will close only with close button
167 				if (that.closable === "button") {
168 					return;
169 				}
170 
171 				// Default Closable behavior
172 				// It will close with click on document, too
173 				that.on("show", function () {
174 					$document.one("click", that.innerHide);
175 				});
176 
177 				// Stop event propatation, if click container.
178 				that.$container.on("click", function (event) {
179 					event.stopPropagation();
180 				});
181 			};

205 		/**
206 		 * Inner function that resolves the component's layout and returns a static reference.
207 		 * @protected
208 		 * @name ch.Floats#$container
209 		 * @type jQuery
210 		 */
211 		that.$container = (function () {
212 
213 			// Final jQuery Object
214 			var $container,
215 
216 			// HTML Div Element with role for WAI-ARIA
217 			container = ["<div role=\"" + conf.aria.role + "\""];
218 
219 			var id = "ch-" + that.type + "-" + that.uid;
220 			// ID for WAI-ARIA
221 			if (ch.util.hasOwn(conf.aria, "identifier")) {
222 				// Generated ID using component name and its instance order
223 
224 				// Add aria attribute to trigger element
225 				that.$element.attr(conf.aria.identifier, id);
226 			}
227 
228 			// Add ID to container element
229 			container.push(" id=\"" + id + "\"");
230 
231 			// Classname with component type and extra classes from conf.classes
232 			container.push(" class=\"ch-hide ch-" + that.type + (ch.util.hasOwn(conf, "classes") ? " " + conf.classes : "") + "\"");
233 
234 			// Z-index
235 			container.push(" style=\"z-index:" + (ch.util.zIndex += 1) + ";");
236 
237 			// Width
238 			if (ch.util.hasOwn(conf, "width")) {
239 				container.push("width:" + conf.width + ((typeof conf.width === "number") ? "px;" : ";"));
240 			}
241 
242 			// Height
243 			if (ch.util.hasOwn(conf, "height")) {
244 				container.push("height:" + conf.height + ((typeof conf.height === "number") ? "px;" : ";"));
245 			}
246 
247 			// Style and tag close
248 			container.push("\">");
249 
250 			// Create cone
251 			if ($html.hasClass("lt-ie8") && ch.util.hasOwn(conf, "cone")) {
252 				container.push("<div class=\"ch-" + that.type + "-cone\"></div>");
253 			}
254 
255 			// Tag close
256 			container.push("</div>");
257 
258 			// jQuery Object generated from string
259 			$container = $(container.join(""));
260 
261 			// Create cone
262 			if (ch.util.hasOwn(conf, "cone")) { $container.addClass("ch-cone"); }
263 
264 			// Efects configuration
265 			conf.fx = ch.util.hasOwn(conf, "fx") ? conf.fx : true;
266 
267 			// Position component configuration
268 			conf.position = conf.position || {};
269 			conf.position.element = $container;
270 			conf.position.reposition = ch.util.hasOwn(conf, "reposition") ? conf.reposition : true;
271 
272 			// Initialize positioner component
273 			that.position = ch.Positioner(conf.position);
274 
275 			// Return the entire Layout
276 			return $container;
277 		})();

288 		/**
289 		 * Inner show method. Attach the component layout to the DOM tree.
290 		 * @protected
291 		 * @function
292 		 * @name ch.Floats#innerShow
293 		 * @returns itself
294 		 */
295 		that.innerShow = function (event) {
296 			if (event) {
297 				ch.util.prevent(event);
298 			}
299 
300 			// Avoid showing things that are already shown
301 			if (that.active) return;
302 
303 			that.active = true;
304 
305 			// Add layout to DOM tree
306 			// Increment zIndex
307 			that.$container
308 				.appendTo("body")
309 				.css("z-index", ch.util.zIndex++);
310 
311 			// Request the content
312 			that.content.set();
313 
314 			/**
315 			 * Triggers when component is visible.
316 			 * @name ch.Floats#show
317 			 * @event
318 			 * @public
319 			 * @exampleDescription It change the content when the component was shown.
320 			 * @example
321 			 * widget.on("show",function () {
322 			 * this.content("Some new content");
323 			 * });
324 			 * @see ch.Floats#show
325 			 */
326 			// Show component with effects
327 			if (conf.fx) {
328 				that.$container.fadeIn("fast", function () {
329 
330 					that.$container.removeClass("ch-hide");
331 					// new callbacks
332 					that.trigger("show");
333 					// Old callback system
334 					that.callbacks('onShow');
335 
336 				});
337 			} else {
338 			// Show component without effects
339 				that.$container.removeClass("ch-hide");
340 				// new callbacks
341 				that.trigger("show");
342 				// Old callback system
343 				that.callbacks('onShow');
344 			}
345 
346 			that.position("refresh");
347 
348 			return that;
349 		};

351 		/**
352 		 * Inner hide method. Hides the component and detach it from DOM tree.
353 		 * @protected
354 		 * @function
355 		 * @name ch.Floats#innerHide
356 		 * @returns itself
357 		 */
358 		that.innerHide = function (event) {
359 
360 			if (event) {
361 				event.stopPropagation();
362 			}
363 
364 			if (!that.active) { return; }
365 
366 			that.active = false;
367 
368 			var afterHide = function () {
369 				/**
370 				 * Triggers when component is not longer visible.
371 				 * @name ch.Floats#hide
372 				 * @event
373 				 * @public
374 				 * @exampleDescription When the component hides show other component.
375 				 * @example
376 				 * widget.on("hide",function () {
377 				 * otherComponent.show();
378 				 * });
379 				 */
380 				// new callbacks
381 				that.trigger("hide");
382 				// Old callback system
383 				that.callbacks('onHide');
384 
385 				that.$container.detach();
386 
387 			};
388 
389 			// Show component with effects
390 			if (conf.fx) {
391 				that.$container.fadeOut("fast", afterHide);
392 
393 			// Show component without effects
394 			} else {
395 				that.$container.addClass("ch-hide");
396 				afterHide();
397 			}
398 
399 			// Removes the innerHide listener
400 			// #708 Modal: The widget closes by itself when It's showing the second time
401 			$document.off("click", that.innerHide);
402 
403 			return that;
404 
405 		};

409 		 * @protected

411 		 * @name ch.Floats#size
412 		 * @param {String} prop Property that will be setted or getted, like "width" or "height".
413 		 * @param {String} [data] Only for setter. It's the new value of defined property.
414 		 * @returns itself

416 		that.size = function (prop, data) {
417 			// Getter
418 			if (!data) { return that.conf[prop]; }
419 
420 			// Setter
421 			that.conf[prop] = data;
422 
423 			// Container size
424 			that.$container[prop](data);
425 
426 			// Refresh position
427 			that.position("refresh");
428 
429 			return that["public"];
430 		};

493 		/**
494 		 * Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
495 		 * @public
496 		 * @function
497 		 * @name ch.Floats#position
498 		 * @exampleDescription Change component's position.
499 		 * @example
500 		 * widget.position({
501 		 *   offset: "0 10",
502 		 *   points: "lt lb"
503 		 * });
504 		 * @exampleDescription Refresh position.
505 		 * @example
506 		 * widget.position("refresh");
507 		 * @see ch.Floats#position
508 		 */
509 		// Create a custom Positioner object to update conf.position data of Float family
510 		that["public"].position = function (o) {
511 
512 			var r = that["public"];
513 
514 			switch (typeof o) {
515 			// Custom Setter: It updates conf.position data
516 			case "object":
517 				// New points
518 				if (ch.util.hasOwn(o, "points")) { conf.position.points = o.points; }
519 
520 				// New reposition
521 				if (ch.util.hasOwn(o, "reposition")) { conf.position.reposition = o.reposition; }
522 
523 				// New offset (splitted)
524 				if (ch.util.hasOwn(o, "offset")) { conf.position.offset = o.offset; }
525 
526 				// New context
527 				if (ch.util.hasOwn(o, "context")) { conf.position.context = o.context; }
528 
529 				// Original Positioner
530 				that.position(conf.position);
531 
532 				break;
533 
534 			// Refresh
535 			case "string":
536 				that.position("refresh");
537 
538 				break;
539 
540 			// Getter
541 			case "undefined":
542 			default:
543 				r = that.position();
544 
545 				break;
546 			}
547 
548 			return r;
549 		};
