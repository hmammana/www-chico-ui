 77         /**
 78         * Create controller's children.
 79         * @private
 80         * @name ch.Tabs#createTabs
 81         * @function
 82         */
 83             createTabs = function () {
 84 
 85                 // Children
 86                 that.$triggers.find("a").each(function (i, e) {
 87 
 88                     // Tab context
 89                     var tab = {};
 90                         tab.type = tab.name = "tab";
 91                         tab.element = e;
 92                         tab.$element = $(e);
 93                         tab.controller = that["public"];
 94 
 95                     // Tab configuration
 96                     var config = {};
 97                         config.open = (selected === i);
 98                         config.onShow = function () { selected = i; };
 99 
100                     if (ch.util.hasOwn(that.conf, "cache")) { config.cache = that.conf.cache; }
101                     if (ch.util.hasOwn(that.conf, "method")) { config.method = that.conf.method; }
102 
103                     /**
104                     * Fired when the content of one dynamic tab loads.
105                     * @name ch.Tabs#contentLoad
106                     * @event
107                     * @public
108                     */
109                     if (ch.util.hasOwn(that.conf, "onContentLoad")) { config.onContentLoad = that.conf.onContentLoad; }
110 
111                     /**
112                     * Fired when the content of one dynamic tab did not load.
113                     * @name ch.Tabs#contentError
114                     * @event
115                     * @public
116                     */
117                     if (ch.util.hasOwn(that.conf, "onContentError")) { config.onContentError = that.conf.onContentError; }
118 
119                     // Create Tabs
120                     that.children.push(ch.Tab.call(tab, config));
121 
122                     // Bind new click to have control
123                     $(e).on("click", function (event) {
124                         ch.util.prevent(event);
125                         select(i);
126                     });
127 
128                 });
129 
130                 return;
131 
132             },

134         /**
135         * Select a child to show its content.
136         * @name ch.Tabs#select
137         * @private
138         * @function
139         */
140             select = function (index) {
141 
142                     // Sets the tab's index
143                 var tab = that.children[index],
144                     regExp;
145 
146                 // If select a tab that doesn't exist do nothing
147                 // Don't click me if I'm open
148                 if (!tab || index === selected) {
149                     return that;
150                 }
151 
152                 // Hides the open tab
153                 if (typeof selected !== "undefined") {
154                     that.children[selected].innerHide();
155                 }
156 
157                 // Shows the current tab
158                 tab.innerShow();
159 
160                 // Updated selected index
161                 selected = index;
162 
163                 // Update window location hash
164                 if (that._currentHash === '') {
165                     that._currentHash = '#!/' + tab.$content.attr("id");
166                 } else {
167                     regExp = new RegExp(window.location.hash.match(/\#!?\/?(.[^\?|\&|\s]+)/)[1]) ;
168                     that._currentHash = window.location.hash.replace(regExp, tab.$content.attr("id"));
169                 }
170 
171                 window.location.hash = that._currentHash;
172 
173                 /**
174                 * Fired when a tab is selected.
175                 * @name ch.Tabs#select
176                 * @event
177                 * @public
178                 */
179                 that.trigger("select");
180 
181                 // Callback
182                 that.callbacks("onSelect");
183 
184                 return that;
185             };

245         * Select a specific tab or get the selected tab.
246         * @public
247         * @name ch.Tabs#select
248         * @function
249         * @param {Number} [tab] Tab's index.
250         * @exampleDescription Selects a specific tab
251         * @example
252         * widget.select(2);
253         * @exampleDescription Returns the selected tab's index
254         * @example
255         * var selected = widget.select();

257         that["public"].select = function (tab) {
258             // Returns selected tab instead set it
259             // Getter
260             if (!parseInt(tab)) {

262             }
263

276         for(var i = that.children.length; i--;) {
277             if (that.children[i].$content.attr("id") === this._currentHash) {
278                 select(i);
279                 hashed = true;
280                 break;
281             }
282         };
283 
284         // Shows the first tab if not hash or it's hash and it isn't from the current tab
285         if( !this._currentHash || ( this._currentHash && !hashed ) ){
286             that.children[0].innerShow();
287             selected = 0;
288         }

336     function Tab(conf) {
337         /**
338         * Reference to a internal component instance, saves all the information and configuration properties.
339         * @private
340         * @name ch.Tab#that
341         * @type object
342         * @ignore
343         */
344         var that = this;
345 
346         conf = ch.util.clone(conf);
347         conf.icon = false;
348 
349         that.conf = conf;
350 
351     /**
352     *   Inheritance
353     */
354         that = ch.Widget.call(that);
355         that.parent = ch.util.clone(that);
356 
357     /**
358      * Abilities
359      */
360 
361         ch.Content.call(that);
362 
363         /**
364          * This callback is triggered when content request have finished.
365          * @protected
366          * @name ch.Floats#onmessage
367          * @function
368          * @returns {this}
369          */
370         that.content.onmessage = function (data) {
371 
372             that.$content.html(data);
373 
374             that.controller.trigger("contentLoad");
375 
376             if (ch.util.hasOwn(conf, "onContentLoad")) {
377                 conf.onContentLoad.call(that.controller, data);
378             }
379         };
380 
381         /**
382          * This callback is triggered when async request fails.
383          * @protected
384          * @name ch.Floats#onerror
385          * @function
386          * @returns {this}
387          */
388         that.content.onerror = function (data) {
389 
390             that.$content.html(data);
391 
392             that.controller.trigger("contentError", data);
393 
394             if (ch.util.hasOwn(conf, "onContentError")) {
395                 conf.onContentError.call(that.controller, data.jqXHR, data.textStatus, data.errorThrown);
396             }
397         };

399     /**
400     *   Private Members
401     */
402         /**
403         * Creates the basic structure for the tab's content.
404         * @private
405         * @name ch.Tab#createContent
406         * @function
407         * @ignore
408         */
409         var createContent = function () {
410 
411             var href = that.element.href.split("#"),
412                 controller = that.$element.parents(".ch-tabs"),
413                 content = controller.find("#" + href[1]),
414                 contentOptions;
415 
416             // If there are a tabContent...
417             if (content.length > 0) {
418                 return content;
419 
420             // If tabContent doesn't exists
421             } else {
422 
423                 /**
424                 * Content configuration property.
425                 * @public
426                 * @name ch.Tab#source
427                 * @type string
428                 * @ignore
429                 */
430                 that.source = href[0];
431 
432                 contentOptions = {
433                     'input': that.source,
434                     'method': that.conf.method,
435                     'cache': that.conf.cache
436                 };
437 
438                 that.content.configure(contentOptions);
439 
440                 var id = (href.length === 2) ? href[1] : "ch-tab-" + that.uid;
441 
442                 // Create tabContent
443                 return $("<div id=\"" + id + "\" role=\"tabpanel\" class=\"ch-hide\">").appendTo(controller.children().eq(1));
444             }
445 
446         };

469         /**
470         * Shows component's content.
471         * @protected
472         * @function
473         * @name ch.Tab#innerShow
474         * @returns itself
475         * @ignore
476         */
477         that.innerShow = function (event) {
478             ch.util.prevent(event);
479 
480             that.active = true;
481 
482             // Load my content if I'need an ajax request
483             if (ch.util.hasOwn(that, "source")) { that.content.set(); }
484 
485             // Show me
486             that.$trigger.addClass("ch-" + that["type"] + "-trigger-on");
487 
488             // Set me as hidden false
489             that.$content
490                 .attr("aria-hidden", "false")
491                 .removeClass("ch-hide");
492 
493             return that;
494         };
495 
496         /**
497         * Hides component's content.
498         * @protected
499         * @function
500         * @name ch.Tab#innerHide
501         * @returns itself
502         * @ignore
503         */
504         that.innerHide = function (event) {
505             ch.util.prevent(event);
506 
507             if (!that.active) { return; }
508 
509             that.active = false;
510 
511             // Hide me
512             that.$trigger.removeClass("ch-" + that["type"] + "-trigger-on");
513 
514             // Set all inactive tabs as hidden
515             that.$content
516                 .attr("aria-hidden", "true")
517                 .addClass("ch-hide");
518 
519             return that;
520         };

529     */
530 
531         // Add the attributes for WAI-ARIA to the tabs and tabpanel
532         // By default is hidden
533         that.$content.attr({
534             "role": "tabpanel",
535             "aria-hidden": true,
536             "class": "ch-hide"
537         });
538 
539         that.$trigger.attr({
540             "role": "tab",
541             "arial-controls": that.$content.attr("id"),
542             "class": "ch-tab-trigger"
543         });
544 
545         return that;
546     }
