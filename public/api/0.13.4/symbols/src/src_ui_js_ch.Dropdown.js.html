1 /**
  2 * Dropdown shows a list of options for navigation.
  3 * @name Dropdown
  4 * @class Dropdown
  5 * @augments ch.Navs
  6 * @requires ch.Positioner
  7 * @see ch.Navs
  8 * @see ch.Positioner
  9 * @see ch.Expando
 10 * @see ch.TabNavigator
 11 * @memberOf ch
 12 * @param {Object} [conf] Object with configuration properties.
 13 * @param {Boolean} [conf.open] Shows the dropdown open when component was loaded. By default, the value is false.
 14 * @param {Boolean} [conf.icon] Shows an arrow as icon. By default, the value is true.
 15 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 16 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 17 * @returns itself
 18 * @factorized
 19 * @exampleDescription Create a new dropdown without configuration.
 20 * @example
 21 * var widget = $(".example").dropdown();
 22 * @exampleDescription Create a new dropdown with configuration.
 23 * @example
 24 * var widget = $(".example").dropdown({
 25 *     "open": true,
 26 *     "icon": false,
 27 *     "points": "lt lt",
 28 *     "fx": true
 29 * });
 30 */

 48 
 49 		that.$element = $el;
 50 		that.element = $el[0];
 51 		that.type = 'dropdown';
 52 		conf = conf ||Â {};
 53 
 54 		conf = ch.util.clone(conf);
 55 
 56 		conf.reposition = ch.util.hasOwn(conf, "reposition") ? conf.reposition : true;
 57 
 58 		that.conf = conf;
 59 
 60 		/**
 61 		 *	Inheritance
 62 		 */
 63 
 64 		that = ch.Navs.call(that);
 65 		that.parent = ch.util.clone(that);

 68 		 *  Private Members

 76 		/**
 77 		 * The component's trigger.
 78 		 * @protected
 79 		 * @name ch.Dropdown#$trigger
 80 		 * @type jQuery
 81 		 */
 82 		that.$trigger = (function () {
 83 
 84 			var $el = that.$trigger;
 85 
 86 			if (!that.$element.hasClass("ch-dropdown-skin")) {
 87 				$el.addClass("ch-btn-skin ch-btn-small");

 92 		}());

 94 		/**
 95 		 * The component's content.
 96 		 * @protected
 97 		 * @name ch.Dropdown#$content
 98 		 * @type jQuery
 99 		 */
100 		that.$content = (function () {
101 
102 			// jQuery Object
103 			var $content = that.$content
104 			// Prevent click on content (except links)
105 				.bind("click", function(event) {
106 					if ((event.target || event.srcElement).tagName === "A") {
107 						that.innerHide();
108 					}
109 					event.stopPropagation();
110 				})
111 			// WAI-ARIA properties
112 				.attr({ "role": "menu", "aria-hidden": "true" });
113 
114 			// WAI-ARIA for options into content
115 			$content.children("a").attr("role", "menuitem");
116 
117 			// Position
118 			that.position = ch.Positioner({
119 				"element": $content,
120 				"context": that.$trigger,
121 				"points": (conf.points || "lt lb"),
122 				"offset": "0 -1",
123 				"reposition": conf.reposition
124 			});
125 
126 			return $content;
127 		}());

131 		 * Dropdown options.
132 		 * @protected

136 		that.$options = that.$content.find("a");
137 
138 
139 		/**
140 		 * Keyboard events object.
141 		 * @protected
142 		 * @Object
143 		 * @name ch.Dropdown#shortcuts
144 		 */
145 		that.shortcuts = (function () {
146 			var selected,
147 				map = {},
148 				arrow,
149 				optionsLength = that.$options.length,
150 				selectOption = function (key, event) {
151 					// Validations
152 					if (!that.active) { return; }
153 
154 					// Prevent default behaivor
155 					ch.util.prevent(event);
156 
157 					// Sets the arrow that user press
158 					arrow = key.type;
159 
160 					// Sets limits behaivor
161 					if (selected === (arrow === "down_arrow" ? optionsLength - 1 : 0)) { return; }
162 
163 					// Unselects current option
164 					that.$options[selected].blur();
165 
166 					if (arrow === "down_arrow") { selected += 1; } else { selected -= 1; }
167 
168 					// Selects new current option
169 					that.$options[selected].focus();
170 				};
171 
172 			return function () {
173 				// Keyboard support initialize
174 				selected = 0;
175 
176 				// Item selected by mouseover
177 				$.each(that.$options, function (i, e) {
178 					$(e).bind("mouseenter", function () {
179 						that.$options[selected = i].focus();
180 					});
181 				});
182 
183 				// Creates keyboard shortcuts map and binding events
184 				map["click " + ch.events.key.ESC] = function () { that.innerHide();Â };
185 				map[ch.events.key.UP_ARROW + " " +ch.events.key.DOWN_ARROW] = selectOption;
186 				$document.on(map);
187 			}
188 		}());
189 
190 		/**
191 		 * Shows component's content.
192 		 * @protected
193 		 * @function
194 		 * @name ch.Dropdown#innerShow
195 		 * @returns itself
196 		 */
197 		that.innerShow = function (event) {
198 
199 			// Stop propagation
200 			ch.util.prevent(event);
201 
202 			// Z-index of content and updates aria values
203 			that.$content.css("z-index", ch.util.zIndex += 1).attr("aria-hidden", "false");
204 
205 			// Z-index of trigger over content (secondary / skin dropdown)
206 			if (that.$element.hasClass("ch-dropdown-skin")) { that.$trigger.css("z-index", ch.util.zIndex += 1); }
207 
208 			// Inheritance innerShow
209 			that.parent.innerShow(event);
210 
211 			// Refresh position
212 			that.position("refresh");
213 
214 			// Reset all dropdowns except itself
215 			$.each(ch.instances.dropdown, function (i, e) {
216 				if (e.uid !== that.uid) { e.hide(); }
217 			});
218 
219 			that.$options[0].focus();
220 
221 			return that;
222 		};
223 
224 		/**
225 		 * Hides component's content.
226 		 * @protected
227 		 * @function
228 		 * @name ch.Dropdown#innerHide
229 		 * @returns itself
230 		 */
231 		that.innerHide = function (event) {
232 
233 			// Inheritance innerHide
234 			that.parent.innerHide(event);
235 
236 			// Updates aria values
237 			that.$content.attr("aria-hidden", "true");
238 
239 			return that;
240 		};
241

247 		 * @borrows ch.Widget#uid as ch.Menu#uid

250 		 * @borrows ch.Navs#show as ch.Dropdown#type

252 		 */

255 		 * Positioning configuration.
256 		 * @public
257 		 * @name ch.Dropdown#position
258 		 * @function
259 		 */
260 		that["public"].position = that.position;
261 
262 		/**
263 		 *  Default event delegation
264 		 */
265 
266 		ch.util.avoidTextSelection(that.$trigger);
267 
268 		// Inits keyboards support
269 		that.shortcuts();
270

273 		 * @name ch.Dropdown#ready
274 		 * @event
275 		 * @public
276 		 * @since 0.8.0
277 		 * @exampleDescription Following the first example, using <code>widget</code> as dropdown's instance controller:
278 		 * @example
279 		 * widget.on("ready",function () {
280 		 *	this.show();
281 		 * });
282 		 */
283 		window.setTimeout(function(){ that.trigger("ready")}, 50);
284 
285 		return that['public'];
