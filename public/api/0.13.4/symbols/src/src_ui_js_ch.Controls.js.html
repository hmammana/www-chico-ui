1 /**
  2 * Controls brings the functionality of all form controls.
  3 * @abstract
  4 * @name Controls
  5 * @class Controls
  6 * @augments ch.Widget
  7 * @requires ch.Floats
  8 * @memberOf ch
  9 * @returns itself
 10 * @see ch.Countdown
 11 * @see ch.Validation
 12 * @see ch.AutoComplete
 13 * @see ch.DatePicker
 14 * @see ch.Widget
 15 * @see ch.Floats
 16 */
 17 (function (window, $, ch) {
 18 	'use strict';
 19 
 20 	if (ch === undefined) {
 21 		throw new window.Error('Expected ch namespace defined.');
 22 	}
 23 
 24 	function Controls($el, conf) {
 25 
 26 		/**
 27 		* Reference to a internal component instance, saves all the information and configuration properties.
 28 		* @name ch.Controls#that
 29 		* @type Object
 30 		*/
 31 		var that = this;
 32 		var	conf = conf || {};
 33 
 34 	/**
 35 	*  Inheritance
 36 	*/
 37 		that = ch.Widget.call(that);
 38 		that.parent = ch.util.clone(that);
 39 
 40 	/**
 41 	*  Protected Members
 42 	*/
 43 
 44 		/**
 45 		* Creates a reference to the Float component instanced.
 46 		* @protected
 47 		* @type Object
 48 		* @name ch.Controls#createFloat
 49 		*/
 50 		that.createFloat = function (c) {
 51 			c.position = {
 52 				"context": conf.context || c.context || c.$element || that.$element,
 53 				"offset": c.offset,
 54 				"points": c.points
 55 			};
 56 
 57 			return ch.Floats.call({
 58 				"element": (ch.util.hasOwn(c, "$element")) ? c.$element[0] : that.element,
 59 				"$element": c.$element || that.$element,
 60 				"source": c.content,
 61 				"uid": (ch.util.index += 1),
 62 				"type": c.type || that.type,
 63 				"conf": c
 64 			});
 65 		};
 66 
 67 	/**
 68 	*  Public Members
 69 	*/
 70 
 71 		return that;
 72 	}
 73 
 74 	ch.Controls = Controls;
 75 
 76 }(this, this.jQuery, this.ch));
