  3 * @name Positioner
  4 * @class Positioner

 10 * @param {String} [conf.offset] Offset in pixels that element will be displaced from original position determined by points. It's specified by configuration or zero by default.
 11 * @param {Boolean} [conf.reposition] Parameter that enables or disables reposition intelligence. It's disabled by default.
 12 * @requires ch.Viewport
 13 * @see ch.Viewport
 14 * @returns {Function} The Positioner returns a Function that it works in 3 ways: as a setter, as a getter and with the "refresh" parameter refreshes the position.

 18 *
 19 * @example
 20 * var positioned = ch.positioner({
 21 *     element: "#element1",
 22 * });
 23 * @exampleDescription 1. Getting the current configuration properties.
 24 * @example

 27 * @example
 28 * positioned("refresh");
 29 * @exampleDescription 3. Define a new position

 33 *     context: "#context2",
 34 *     points: "lt rt"
 35 * });

443 		/**
444 		 * Calculates left and top position from specific points.
445 		 * @private
446 		 * @name ch.Positioner#getCoordinates
447 		 * @function
448 		 * @param {String} points String with points to be calculated.
449 		 * @returns Offset measures
450 		 * @exampleDescription
451 		 * @example
452 		 * var foo = getCoordinates("lt rt");
453 		*
454 		 * foo = {
455 		 *     left: Number,
456 		 *     top: Number
457 		 * };
458 		 */
459 			getCoordinates = function (pts) {
460 
461 				// Calculates left or top position from points related to specific axis (X or Y).
462 				// TODO: Complete cases: X -> lc, cl, rc, cr. Y -> tm, mt, bm, mb.
463 				var calculate = function (reference) {
464 
465 					// Use Position or Offset of Viewport if position is fixed or absolute respectively
466 					var ctx = (!contextIsNotViewport && ch.support.fixed) ? ch.viewport.getPosition() : context,
467 
468 					// Returnable value
469 						r;
470 
471 					switch (reference) {
472 					// X references
473 					case "ll": r = ctx.left + offset[0]; break;
474 					case "lr": r = ctx.right + offset[0]; break;
475 					case "rl": r = ctx.left - $element.outerWidth() + offset[0]; break;
476 					case "rr": r = ctx.right - $element.outerWidth() + offset[0]; break;
477 					case "cc": r = ctx.left + (ctx.width / 2) - ($element.outerWidth() / 2) + offset[0]; break;
478 					// Y references
479 					case "tt": r = ctx.top + offset[1]; break;
480 					case "tb": r = ctx.bottom + offset[1]; break;
481 					case "bt": r = ctx.top - $element.outerHeight() + offset[1]; break;
482 					case "bb": r = ctx.bottom - $element.outerHeight() + offset[1]; break;
483 					case "mm": r = ctx.top + (ctx.height / 2) - ($element.outerHeight() / 2) + offset[1]; break;
484 					}
485 
486 					return r;
487 				},
488 
489 				// Splitted points
490 					splittedPoints = pts.split(" ");
491 
492 				// Calculates left and top with references to X and Y axis points (crossed points)
493 				return {
494 					"left": calculate(splittedPoints[0].charAt(0) + splittedPoints[1].charAt(0)),
495 					"top": calculate(splittedPoints[0].charAt(1) + splittedPoints[1].charAt(1))
496 				};
497 			},
