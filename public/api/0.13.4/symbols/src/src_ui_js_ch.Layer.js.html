1 /**
  2  * Layer lets you show a contextual floated data.
  3  * @name Layer
  4  * @class Layer
  5  * @augments ch.Floats
  6  * @standalone
  7  * @memberOf ch
  8  * @param {Object} [conf] Object with configuration properties.
  9  * @param {String} [conf.content] Sets content by: static content, DOM selector or URL. By default, the content is empty.
 10  * @param {Number|String} [conf.width] Sets width property of the component's layout. By default, the width is "500px".
 11  * @param {Number|String} [conf.height] Sets height property of the component's layout. By default, the height is elastic.
 12  * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
 13  * @param {String} [conf.event] Sets the event ("click" or "hover") that trigger show method. By default, the event is "hover".
 14  * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 15  * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 16  * @param {Boolean} [conf.cache] Enable or disable the content cache. By default, the cache is enable.
 17  * @param {String} [conf.closable] Sets the way (true, "button" or false) the Layer close when conf.event is set as "click". By default, the layer close true.
 18  * @returns itself
 19  * @factorized
 20  * @see ch.Floats
 21  * @see ch.Tooltip
 22  * @see ch.Modal
 23  * @see ch.Zoom
 24  * @exampleDescription To create a ch.Layer you have to give a selector.
 25  * @example
 26  * var widget = $(".some-element").layer("<tag>Some content.</tag>");
 27  * @exampleDescription ch.Layer component can receive a parameter. It is a literal object { }, with the properties you want to configurate.
 28  * @example
 29  * var conf = {
 30  *     "width": 200,
 31  *     "height": 50
 32  * };
 33  * @exampleDescription Create a layer with configuration.
 34  * @example
 35  * var widget = $(".some-element").layer({
 36  *     "content": "Some content here!",
 37  *     "width": "200px",
 38  *     "height": 50,
 39  *     "event": "click",
 40  *     "closable": "button",
 41  *     "offset": "10 -10",
 42  *     "cache": false,
 43  *     "points": "lt rt"
 44  * });
 45  * @exampleDescription Now <code>widget</code> is a reference to the layer instance controller. You can set a new content by using <code>widget</code> like this:
 46  * @example
 47  * widget.content("http://content.com/new/content");
 48  */
 49 (function (window, $, ch) {
 50 	'use strict';
 51 
 52 	if (ch === undefined) {
 53 		throw new window.Error('Expected ch namespace defined.');
 54 	}
 55 
 56 	function Layer($el, conf) {
 57 
 58 		/**
 59 		 * Reference to a internal component instance, saves all the information and configuration properties.
 60 		 * @private
 61 		 * @name ch.Layer#that
 62 		 * @type object
 63 		 */
 64 		var that = this;
 65 		that.$element = $el;
 66 		that.element = $el[0];
 67 		that.type = 'layer';
 68 		conf = conf || {};
 69 
 70 		conf = ch.util.clone(conf);
 71 
 72 		conf.cone = true;
 73 		conf.classes = conf.classes || "ch-box-lite";
 74 
 75 		// Closable configuration
 76 		conf.closeButton = ch.util.hasOwn(conf, "closeButton") ? conf.closeButton : (conf.event === "click");
 77 		conf.closable = ch.util.hasOwn(conf, "closable") ? conf.closable : true;
 78 
 79 		conf.aria = {};
 80 		conf.aria.role = "tooltip";
 81 		conf.aria.identifier = "aria-describedby";
 82 
 83 		conf.position = {};
 84 		conf.position.context = that.$element;
 85 		conf.position.offset = conf.offset || "0 10";
 86 		conf.position.points = conf.points || "lt lb";
 87 
 88 		that.conf = conf;
 89 
 90 		/**
 91 		 * Content configuration property.
 92 		 * @protected
 93 		 * @name ch.Modal#source
 94 		 */
 95 		that.source = conf.content;
 96 	/**
 97 	 * Inheritance
 98 	 */
 99 
100 		that = ch.Floats.call(that);
101 		that.parent = ch.util.clone(that);
102 
103 	/**
104 	 * Private Members
105 	 */
106 
107 		/**
108 		 * Delay time to hide component's contents.
109 		 * @private
110 		 * @name ch.Layer#hideTime
111 		 * @type number
112 		 * @default 400
113 		 */
114 		var hideTime = 400,
115 
116 		/**
117 		 * Hide timer instance.
118 		 * @private
119 		 * @name ch.Layer#ht
120 		 * @type timer
121 		 */
122 			ht,
123 
124 		/**
125 		 * Starts hide timer.
126 		 * @private
127 		 * @function
128 		 * @name ch.Layer#hideTimer
129 		 */
130 			hideTimer = function (event) {
131 				if (conf.event !== "click") {
132 					var target = event.srcElement || event.target;
133 
134 					var relatedTarget = event.relatedTarget || event.toElement;
135 
136 					if (relatedTarget === null || target === relatedTarget || relatedTarget === undefined || relatedTarget.parentNode === null || target.nodeName === "SELECT") {
137 						return;
138 					}
139 				}
140 
141 				ht = setTimeout(function () { that.innerHide() }, hideTime);
142 			},
143 
144 		/**
145 		 * Clear all timers.
146 		 * @private
147 		 * @function
148 		 * @name ch.Layer#clearTimers
149 		 */
150 			clearTimers = function () { clearTimeout(ht); };
151 
152 	/**
153 	 * Protected Members
154 	 */
155 
156 		/**
157 		 * Inner show method. Attach the component layout to the DOM tree.
158 		 * @protected
159 		 * @function
160 		 * @name ch.Layer#innerShow
161 		 * @returns itself
162 		 */
163 		that.innerShow = function (event) {
164 			// Reset all layers, except me and not auto closable layers
165 			$.each(ch.instances.layer, function (i, e) {
166 				if (e !== that["public"] && e.closable() === true) {
167 					e.hide();
168 				}
169 			});
170 
171 			// conf.position.context = that.$element;
172 			that.parent.innerShow(event);
173 
174 			if (conf.event !== "click") {
175 				that.$container.one("mouseenter", clearTimers).bind("mouseleave", hideTimer);
176 			}
177 
178 			return that;
179 		};
180 
181 		/**
182 		 * Inner hide method. Hides the component and detach it from DOM tree.
183 		 * @protected
184 		 * @function
185 		 * @name ch.Layer#innerHide
186 		 * @returns itself
187 		 */
188 		that.innerHide = function (event) {
189 			that.$container.unbind("mouseleave", hideTimer);
190 
191 			that.parent.innerHide(event);
192 		}
193 
194 	/**
195 	 * Public Members
196 	 */
197 
198 		/**
199 		 * @borrows ch.Widget#on as ch.Layer#on
200 		 */
201 
202 
203 	/**
204 	 * Default event delegation
205 	 */
206 
207 		// Click
208 		if (conf.event === "click") {
209 			that.$element
210 				.css("cursor", "pointer")
211 				.bind("click", that.innerShow);
212 
213 		// Hover
214 		} else {
215 			that.$element
216 				.css("cursor", "default")
217 				.bind("mouseenter", that.innerShow)
218 				.bind("mouseleave", hideTimer);
219 		}
220 
221 		/**
222 		 * Triggers when the component is ready to use (Since 0.8.0).
223 		 * @name ch.Layer#ready
224 		 * @event
225 		 * @public
226 		 * @since 0.8.0
227 		 * @exampleDescription Following the first example, using <code>widget</code> as layer's instance controller:
228 		 * @example
229 		 * widget.on("ready",function () {
230 		 * this.show();
231 		 * });
232 		 */
233 		window.setTimeout(function(){ that.trigger("ready")}, 50);
234 
235 		return that['public'];
236 	}
237 
238 	Layer.prototype.name = 'layer';
239 	Layer.prototype.constructor = Layer;
240 
241 	ch.factory(Layer);
242 
243 }(this, this.jQuery, this.ch));
