1 /**
  2  * OnImagesLoads executes a callback function when the images of a query selection loads.
  3  * @name onImagesLoads
  4  * @class onImagesLoads
  5  * @memberOf ch
  6  * @param callback function The function that the component will fire after the images load.
  7  * @returns jQuery
  8  * @factorized
  9  * @exampleDescription
 10  * @example
 11  * $("img").onImagesLoads(callback);
 12  */
 13 (function (window, $, ch) {
 14 	'use strict';
 15 
 16 	if (ch === undefined) {
 17 		throw new window.Error('Expected ch namespace defined.');
 18 	}
 19 
 20 	function OnImagesLoads($el, conf) {
 21 
 22 		/**
 23 		 * Reference to a internal component instance, saves all the information and configuration properties.
 24 		 * @private
 25 		 * @name ch.onImagesLoads#that
 26 		 * @type object
 27 		 */
 28 		var that = this;
 29 		that.$element = $el;
 30 		that.element = $el[0];
 31 		that.type = 'onImagesLoads';
 32 		conf = conf || {};
 33 
 34 		conf = ch.util.clone(conf);
 35 		that.conf = conf;
 36 
 37 		that.$element
 38 			// On load event
 39 			.one("load", function () {
 40 				window.setTimeout(function () {
 41 					if (--that.$element.length <= 0) {
 42 						that.conf.fn.call(that.$element, this);
 43 					}
 44 				}, 200);
 45 			})
 46 			// For each image
 47 			.each(function () {
 48 				// Cached images don't fire load sometimes, so we reset src.
 49 				if (this.complete || this.complete === undefined) {
 50 					var src = this.src;
 51 
 52 					// Data uri fix bug in web-kit browsers
 53 					this.src = "data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///ywAAAAAAQABAAACAUwAOw==";
 54 					this.src = src;
 55 				}
 56 			});
 57 
 58 		return that;
 59 	}
 60 
 61 	OnImagesLoads.prototype.name = 'onImagesLoads';
 62 	OnImagesLoads.prototype.constructor = OnImagesLoads;
 63 
 64 	ch.factory(OnImagesLoads);
 65 
 66 }(this, this.jQuery, this.ch));
