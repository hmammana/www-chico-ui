210 		/**
211 		* Runs all validations to check if it has an error.
212 		* @protected
213 		* @type function
214 		* @returns boolean
215 		* @name ch.Validation#process
216 		*/
217 		that.process = function (evt) {
218 
219 			// Pre-validation: Don't validate disabled
220 			if (that.$element.attr("disabled") || !that.enabled) { return false; }
221 
222 			/**
223 			* Triggers before start validation process.
224 			* @name ch.Validation#beforeValidate
225 			* @event
226 			* @public
227 			* @exampleDescription
228 			* @example
229 			* widget.on("beforeValidate",function(event) {
230 			*	submitButton.disable();
231 			* });
232 			*/
233 			// old callback system
234 			that.callbacks('beforeValidate');
235 			// new callback
236 			that.trigger("beforeValidate");
237 
238 			// Executes the validators engine with a specific value and returns an object.
239 			// Context is the validation
240 			var gotError = validator.validate.call(that["public"], that.element.value);
241 
242 			// Save the validator's status.
243 			var status = !gotError.status;
244 
245 			// If has Error...
246 			if (status) {
247 
248 				if (that.$element.prop("tagName") === "INPUT" || that.$element.prop("tagName") === "TEXTAREA") {
249 					// TODO: remove error class when deprecate old forms only ch-form error must be.
250 					that.$element.addClass("ch-form-error");
251 				}
252 
253 				// to avoid reload the same content
254 				if (!that["float"]["public"].isActive() || !that.error.condition || that.error.condition !== gotError.condition) {
255 					that["float"]["public"].show((gotError.msg || form.messages[gotError.condition] || "Error"));
256 					// the aria-label attr should get the message element id, but is not public
257 					that.$element.attr('aria-label', 'ch-' + that["float"]["public"].type + '-' + that["float"]["public"].uid );
258 				}
259 
260 				// Add blur or change event only one time to the element or to the elements's group
261 				if (!that.$element.data("events")) { that.$element.one(validationEvent, function(evt){that.process(evt);}); }
262 
263 				/**
264 				* Triggers when an error occurs on the validation process.
265 				* @name ch.Validation#error
266 				* @event
267 				* @public
268 				* @exampleDescription
269 				* @example
270 				* widget.on("error",function(event, condition) {
271 				*	if (condition === "required") {
272 				* 		errorModal.show();
273 				* 	}
274 				* });
275 				*/
276 				// old callback system
277 				that.callbacks('onError', gotError.condition);
278 				// new callback
279 				that.trigger("error", gotError.condition);
280 
281 				// Saves gotError
282 				that.error = gotError;
283 
284 			// else NOT Error!
285 			} else {
286 				that.$element.removeClass("ch-form-error");
287 				that.$element.removeAttr('aria-label');
288 				that["float"].innerHide();
289 				form.trigger('validated');
290 			}
291 
292 			/**
293 			* Triggers when the validation process ends.
294 			* @name ch.Validation#afterValidate
295 			* @event
296 			* @public
297 			* @exampleDescription
298 			* @example
299 			* widget.on("afterValidate",function(){
300 			*	submitButton.disable();
301 			* });
302 			*/
303 			// old callback system
304 			that.callbacks('afterValidate');
305 			// new callback
306 			that.trigger("afterValidate");
307 
308 			return status;
309 
310 		};

531 		/**

533 		* @public
534 		* @since 0.10.4
535 		* @name ch.Validation#message
536 		* @function
537 		* @returns itself
538 		* @exampleDescription Sets a new message
539 		* @example
540 		* validation.message("required", "New message for required validation");
541 		* @exampleDescription Gets a message from a condition
542 		* @example
543 		* validation.message("required");
544 		*/
545 		that["public"].message = function (condition, msg) {
546 			if (condition === undefined) {
547 				throw "validation.message(condition, message): Please, give me a condition as parameter.";
548 			}
549 
550 			// Get a new message from a condition
551 			if (msg === undefined) {
552 				return validator.conditions[condition].message;
553 			}
554 
555 			// Sets a new message
556 			validator.conditions[condition].message = msg;
557 
558 			// TODO: check if the conditions seted is active
559 			if (validator.isActive()) {
560 
561 				that["public"]["float"].content(msg);
562 			}
563 
564 			return that["public"];
565 		}
