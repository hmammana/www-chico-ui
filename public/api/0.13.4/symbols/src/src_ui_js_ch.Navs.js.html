1 /**
  2 * Navs is a representation of navs components.
  3 * @abstract
  4 * @name Navs
  5 * @class Navs
  6 * @standalone
  7 * @augments ch.Widget
  8 * @memberOf ch
  9 * @param {object} conf Object with configuration properties
 10 * @returns itself
 11 * @see ch.Widget
 12 * @see ch.Dropdown
 13 * @see ch.Expando
 14 */
 15 (function (window, $, ch) {
 16 	'use strict';
 17 
 18 	if (window.ch === undefined) {
 19 		throw new window.Error('Expected ch namespace defined.');
 20 	}
 21 
 22 	var $html = $('html');
 23 
 24 	function Navs($el, conf) {
 25 		/**
 26 		 * Reference to a internal component instance, saves all the information and configuration properties.
 27 		 * @private
 28 		 * @name ch.Navs#that
 29 		 * @type object
 30 		 */
 31 		var that = this,
 32 			conf = that.conf;
 33 
 34 		conf.icon = ch.util.hasOwn(conf, "icon") ? conf.icon : true;
 35 		conf.open = conf.open || false;
 36 		conf.fx = conf.fx || false;
 37 
 38 		/**
 39 		 *	Inheritance
 40 		 */
 41 
 42 		that = ch.Widget.call(that);
 43 		that.parent = ch.util.clone(that);
 44 
 45 		/**
 46 		 *	Protected Members
 47 		 */
 48 
 49 		/**
 50 		 * Status of component
 51 		 * @protected
 52 		 * @name ch.Navs#active
 53 		 * @returns boolean
 54 		 */
 55 		that.active = false;
 56 
 57 		/**
 58 		 * The component's trigger.
 59 		 * @private
 60 		 * @name ch.Navs#$trigger
 61 		 * @type jQuery
 62 		 */
 63 		that.$trigger = that.$element.children().eq(0);
 64 
 65 		/**
 66 		 * The component's content.
 67 		 * @private
 68 		 * @name ch.Navs#$content
 69 		 * @type jQuery
 70 		 */
 71 		that.$content = that.$element.children().eq(1);
 72 
 73 		/**
 74 		 * Shows component's content.
 75 		 * @protected
 76 		 * @name ch.Navs#innerShow
 77 		 * @returns itself
 78 		 */
 79 		that.innerShow = function (event) {
 80 			ch.util.prevent(event);
 81 
 82 			if (that.active) {
 83 				return that.innerHide(event);
 84 			}
 85 
 86 			that.active = true;
 87 
 88 			that.$trigger.addClass("ch-" + that["type"] + "-trigger-on");
 89 
 90 			/**
 91 			 * onShow callback function
 92 			 * @name ch.Navs#onShow
 93 			 * @event
 94 			 */
 95 			// Animation
 96 			if (conf.fx) {
 97 				that.$content.slideDown("fast", function () {
 98 					// new callbacks
 99 					that.trigger("show");
100 					// old callback system
101 					that.callbacks("onShow");
102 				});
103 			} else {
104 				// new callbacks
105 				that.trigger("show");
106 				// old callback system
107 				that.callbacks("onShow");
108 			}
109 
110 			that.$content.removeClass("ch-hide");
111 
112 			return that;
113 		};
114 
115 		/**
116 		 * Hides component's content.
117 		 * @protected
118 		 * @function
119 		 * @name ch.Navs#innerHide
120 		 * @returns itself
121 		 */
122 		that.innerHide = function (event) {
123 			ch.util.prevent(event);
124 
125 			if (!that.active) { return; }
126 
127 			that.active = false;
128 
129 			that.$trigger.removeClass("ch-" + that["type"] + "-trigger-on");
130 
131 			/**
132 			 * onHide callback function
133 			 * @name ch.Navs#onHide
134 			 * @event
135 			 */
136 			// Animation
137 			if (conf.fx) {
138 				that.$content.slideUp("fast", function () {
139 					that.callbacks("onHide");
140 				});
141 			} else {
142 				// new callbacks
143 				that.trigger("hide");
144 				// old callback system
145 				that.callbacks("onHide");
146 			}
147 
148 			that.$content.addClass("ch-hide");
149 
150 			return that;
151 		};
152 
153 		/**
154 		 * Create component's layout
155 		 * @protected
156 		 * @function
157 		 * @name ch.Navs#configBehavior
158 		 */
159 		that.configBehavior = function () {
160 			that.$trigger
161 				.addClass("ch-" + that.type + "-trigger")
162 				.bind("click", function (event) { that.innerShow(event); });
163 
164 			that.$content.addClass("ch-" + that.type + "-content ch-hide");
165 
166 			// Icon configuration
167 			if ($html.hasClass("lt-ie8") && conf.icon) {
168 				$("<span class=\"ch-" + that.type + "-ico\">Drop</span>").appendTo(that.$trigger);
169 			} else if (conf.icon) {
170 				that.$trigger.addClass("ch-" + that.type + "-ico");
171 			}
172 
173 			if (conf.open) { that.innerShow(); }
174 
175 		};
176 
177 		/**
178 		 * Public Members
179 		 */
180 
181 		/**
182 		 * Shows component's content.
183 		 * @public
184 		 * @function
185 		 * @name ch.Navs#show
186 		 * @returns itself
187 		 */
188 		that["public"].show = function(){
189 			that.innerShow();
190 			return that["public"];
191 		};
192 
193 		/**
194 		 * Hides component's content.
195 		 * @public
196 		 * @function
197 		 * @name ch.Navs#hide
198 		 * @returns itself
199 		 */
200 		that["public"].hide = function(){
201 			that.innerHide();
202 			return that["public"];
203 		};
204 
205 		/**
206 		 * Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
207 		 * @public
208 		 * @function
209 		 * @name ch.Navs#isActive
210 		 * @returns boolean
211 		 */
212 		that["public"].isActive = function () {
213 			return that.active;
214 		};
215 
216 		/**
217 		 *	Default event delegation
218 		 */
219 
220 		that.configBehavior();
221 		that.$element.addClass("ch-" + that.type);
222 
223 		/**
224 		 * Triggers when component is visible.
225 		 * @name ch.Navs#show
226 		 * @event
227 		 * @public
228 		 * @example
229 		 * widget.on("show",function () {
230 		 *	otherComponent.hide();
231 		 * });
232 		 * @see ch.Navs#event:show
233 		 */
234 
235 		/**
236 		 * Triggers when component is not longer visible.
237 		 * @name ch.Navs#hide
238 		 * @event
239 		 * @public
240 		 * @example
241 		 * widget.on("hide",function () {
242 		 *	otherComponent.show();
243 		 * });
244 		 * @see ch.Navs#event:hide
245 		 */
246 
247 		return that;
248 
249 	}
250 
251 	Navs.prototype.name = 'navs';
252 	Navs.prototype.constructor = Navs;
253 
254 	ch.Navs = Navs;
255 
256 }(this, this.jQuery, this.ch));
